46/// Goto some position ((1,1)-based).
47///
48/// # Why one-based?
49///
50/// ANSI escapes are very poorly designed, and one of the many odd aspects is being one-based. This
51/// can be quite strange at first, but it is not that big of an obstruction once you get used to
52/// it.
53///
54/// # Example
55///
56/// ```rust
57/// extern crate sl_console;
58///
59/// fn main() {
60///     print!("{}{}Stuff", sl_console::clear::All, sl_console::cursor::Goto(5, 3));
61/// }
62/// ```
63#[derive(Copy, Clone, PartialEq, Eq)]
64pub struct Goto(pub u16, pub u16);
65
66impl From<Goto> for String {
67    fn from(this: Goto) -> String {
68        let (mut x, mut y) = ([0u8; 20], [0u8; 20]);
69        [
70            "\x1B[",
71            this.1.numtoa_str(10, &mut x),
72            ";",
73            this.0.numtoa_str(10, &mut y),
74            "H",
75        ]
76        .concat()
77    }
78}
79
80impl Default for Goto {
81    fn default() -> Goto {
82        Goto(1, 1)
83    }
84}
85
86impl fmt::Display for Goto {
87    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
88        debug_assert!(self != &Goto(0, 0), "Goto is one-based.");
89        write!(f, "\x1B[{};{}H", self.1, self.0)
90    }
91}

161/// Move the cursor to (x, y).
162///
163/// This a convience wrapper.
164pub fn goto(x: u16, y: u16) -> io::Result<()> {
165    let mut conout = conout_r()?.lock();
166    write!(conout, "{}", Goto(x, y))?;
167    conout.flush()?;
168    Ok(())
169}

171/// Return the current cursor position.
172pub fn cursor_pos() -> io::Result<(u16, u16)> {
173    let delimiter = b'R';
174
175    {
176        let mut conout = conout_r()?.lock();
177        // Where is the cursor?
178        // Use `ESC [ 6 n`.
179        write!(conout, "\x1B[6n")?;
180        conout.flush()?;
181    }
182
183    let mut conin = conin_r()?.lock();
184    let mut buf: [u8; 1] = [0];
185    let mut read_chars = Vec::new();
186
187    let timeout = Duration::from_millis(CONTROL_SEQUENCE_TIMEOUT);
188    let now = Instant::now();
189    while buf[0] != delimiter && now.elapsed() < timeout {
190        match conin.read_timeout(&mut buf, Some(timeout - now.elapsed())) {
191            Ok(1) => {
192                read_chars.push(buf[0]);
193            }
194            Ok(0) => {
195                return Err(io::Error::new(
196                    io::ErrorKind::UnexpectedEof,
197                    "Unexpected EOF.",
198                ));
199            }
200            Ok(_) => {}
201            Err(err) if err.kind() == io::ErrorKind::WouldBlock => {}
202            Err(err) => return Err(err),
203        }
204    }
205
206    if read_chars.pop().unwrap_or(b'\0') == delimiter && !read_chars.is_empty() {
207        // The answer will look like `ESC [ Cy ; Cx R`.
208        // The pop in the if removes and verifies the delimiter
209        if let Ok(read_str) = String::from_utf8(read_chars) {
210            if let Some(beg) = read_str.rfind('[') {
211                let coords: String = read_str.chars().skip(beg + 1).collect();
212                let mut nums = coords.split(';');
213                if let (Some(cy), Some(cx)) = (nums.next(), nums.next()) {
214                    if let (Ok(cy), Ok(cx)) = (cy.parse::<u16>(), cx.parse::<u16>()) {
215                        return Ok((cx, cy));
216                    }
217                }
218            }
219        }
220        return Err(Error::new(
221            ErrorKind::Other,
222            "Failed to parse coords from chars read from console.",
223        ));
224    }
225    Err(Error::new(
226        ErrorKind::Other,
227        "Cursor position detection timed out.",
228    ))
229}

231/// Hide the cursor for the lifetime of this struct.
232/// It will hide the cursor on creation with from() and show it back on drop().
233pub struct HideCursor<W: ConsoleWrite> {
234    /// The output target.
235    output: W,
236}
237
238impl<W: ConsoleWrite> HideCursor<W> {
239    /// Create a hide cursor wrapper struct for the provided output and hides the cursor.
240    pub fn from(mut output: W) -> Self {
241        write!(output, "{}", Hide).expect("hide the cursor");
242        HideCursor { output }
243    }
244}
245
246impl<W: ConsoleWrite> Drop for HideCursor<W> {
247    fn drop(&mut self) {
248        write!(self, "{}", Show).expect("show the cursor");
249    }
250}
251
252impl<W: ConsoleWrite> ops::Deref for HideCursor<W> {
253    type Target = W;
254
255    fn deref(&self) -> &W {
256        &self.output
257    }
258}
259
260impl<W: ConsoleWrite> ops::DerefMut for HideCursor<W> {
261    fn deref_mut(&mut self) -> &mut W {
262        &mut self.output
263    }
264}
265
266impl<W: ConsoleWrite> Write for HideCursor<W> {
267    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
268        self.output.write(buf)
269    }
270
271    fn flush(&mut self) -> io::Result<()> {
272        self.output.flush()
273    }
274}
275
276impl<W: ConsoleWrite> ConsoleWrite for HideCursor<W> {
277    fn set_raw_mode(&mut self, mode: bool) -> io::Result<bool> {
278        self.output.set_raw_mode(mode)
279    }
280
281    fn is_raw_mode(&self) -> bool {
282        self.output.is_raw_mode()
283    }
