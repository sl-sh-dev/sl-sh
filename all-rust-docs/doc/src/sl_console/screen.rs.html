1//! Managing switching between main and alternate screen buffers.
2//!
3//! Note that this implementation uses xterm's new escape sequences for screen switching and thus
4//! only works for xterm compatible terminals (which should be most terminals nowadays).
5//!
6//! # Example
7//!
8//! ```rust
9//! use sl_console::conout;
10//! use sl_console::screen::AlternateScreen;
11//! use std::io::{Write, stdout};
12//!
13//!     {
14//!         let mut screen = AlternateScreen::from(conout());
15//!         write!(screen, "Writing to alternate screen!").unwrap();
16//!         screen.flush().unwrap();
17//!     }
18//!     println!("Writing to main screen.");
19//! ```
20
21use std::fmt;
22use std::io::{self, Write};
23use std::ops;
24
25use crate::console::ConsoleWrite;
26
27/// Switch to the main screen buffer of the terminal.
28pub struct ToMainScreen;
29
30impl fmt::Display for ToMainScreen {
31    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
32        write!(f, csi!("?1049l"))
33    }
34}
35
36/// Switch to the alternate screen buffer of the terminal.
37pub struct ToAlternateScreen;
38
39impl fmt::Display for ToAlternateScreen {
40    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
41        write!(f, csi!("?1049h"))
42    }
43}
44
45/// A terminal restorer, which wraps a type implementing Write, and causes all writes to be written
46/// to an alternate screen.
47///
48/// This is achieved by switching the terminal to the alternate screen on creation and
49/// automatically switching it back to the original screen on drop.
50pub struct AlternateScreen<W: ConsoleWrite> {
51    /// The output target.
52    output: W,
53}
54
55impl<W: ConsoleWrite> AlternateScreen<W> {
56    /// Create an alternate screen wrapper struct for the provided output and switch the terminal
57    /// to the alternate screen.
58    pub fn from(mut output: W) -> Self {
59        write!(output, "{}", ToAlternateScreen).expect("switch to alternate screen");
60        AlternateScreen { output }
61    }
62}
63
64impl<W: ConsoleWrite> Drop for AlternateScreen<W> {
65    fn drop(&mut self) {
66        write!(self, "{}", ToMainScreen).expect("switch to main screen");
67    }
68}
69
70impl<W: ConsoleWrite> ops::Deref for AlternateScreen<W> {
71    type Target = W;
72
73    fn deref(&self) -> &W {
74        &self.output
75    }
76}
77
78impl<W: ConsoleWrite> ops::DerefMut for AlternateScreen<W> {
79    fn deref_mut(&mut self) -> &mut W {
80        &mut self.output
81    }
82}
83
84impl<W: ConsoleWrite> Write for AlternateScreen<W> {
85    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
86        self.output.write(buf)
87    }
88
89    fn flush(&mut self) -> io::Result<()> {
90        self.output.flush()
91    }
92}
93
94impl<W: ConsoleWrite> ConsoleWrite for AlternateScreen<W> {
95    fn set_raw_mode(&mut self, mode: bool) -> io::Result<bool> {
96        self.output.set_raw_mode(mode)
97    }
98
99    fn is_raw_mode(&self) -> bool {
100        self.output.is_raw_mode()
101    }
102}
