10// Support functions for converting libc return values to io errors {
11trait IsMinusOne {
12    fn is_minus_one(&self) -> bool;
13}
14
15macro_rules! impl_is_minus_one {
16        ($($t:ident)*) => ($(impl IsMinusOne for $t {
17            fn is_minus_one(&self) -> bool {
18                *self == -1
19            }
20        })*)
21    }
22
23impl_is_minus_one! { i8 i16 i32 i64 isize }
24
25fn cvt<T: IsMinusOne>(t: T) -> io::Result<T> {
26    if t.is_minus_one() {
27        Err(io::Error::last_os_error())
28    } else {
29        Ok(t)
30    }
31}
