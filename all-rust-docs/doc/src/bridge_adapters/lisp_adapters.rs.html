1//! Notes on how conversion works:
2//!
3//! # Accepting argumets
4//! Check rust docs for [`SlFrom`], [`SlFromRef`], and [`SlFromRefMut`] to get a feel for
5//! which rust types will work as parameters to the procedural macro.
6//!
7//! # Returning Arguments
8//! [`SlInto`], [`SlIntoRef`], and [`SlIntoRefMut`] are all inverses of their respective from implementations
9//! and are used to transform the rust types tha function return into slosh values from the procedural macro.
10//!
11//! # On accepting arguments from a dynamically typed language to a static one
12//! Helper types in this crate like [`LooseString`] use the rust type system to model various slosh
13//! types to avoid accepting a raw slosh [`Value`] as a parameter in the procedural macro. This is
14//! not necessary, however, and should only be done if one finds oneself constantly accepting a
15//! [`Value`] type in multiple procedural macro functions and functionally duplicating similar
16//! parsing code.
17//!
18//! # On returning arguments from a statically typed language to a dynamically one
19//! It is also possible to manually return a [`VMResult`] of [`Ok`]<[`Value`]> if the return
20//! type of the function should not be bound by strict typing.
21//!
22//! # Borrowing and Mutable Borrowing when a rust type does not implement [`Copy`], demonstration with &str and &mut String
23//!
24//! [`SlAsRef`] and [`SlAsMut`] can be used to avoid cloning from the [`SloshVm`] into the scope of
25//! function generated by the procedural macro but the traits must be structured properly.
26//!
27//!
28//! ## For regular borrows, copy this pattern used for &str
29//!
30//! ```ignore
31//!
32//! impl<'a> SlFromRef<'a, Value> for &'a str {
33//!     fn sl_from_ref(value: Value, vm: &'a SloshVm) -> VMResult<Self> {
34//!         (&value).sl_as_ref(vm)
35//!     }
36//! }
37//!
38//! impl<'a> SlAsRef<'a, str> for &Value {
39//!     fn sl_as_ref(&self, vm: &'a SloshVm) -> VMResult<&'a str> {
40//!         match self {
41//!             Value::String(h) => Ok(vm.get_string(*h)),
42//!             Value::StringConst(i) => Ok(vm.get_interned(*i)),
43//!             _ => Err(VMError::new_conversion(
44//!                 ErrorStrings::fix_me_mismatched_type(
45//!                     String::from(ValueTypes::from([
46//!                         ValueType::String,
47//!                         ValueType::StringConst,
48//!                     ])),
49//!                     self.display_type(vm),
50//!                 ),
51//!             )),
52//!         }
53//!     }
54//! }
55//! ```
56//!
57//!
58//! # Mutable borrowing when a rust type does not implement copy
59//!
60//! ```ignore
61//!
62//! impl<'a> SlFromRefMut<'a, Value> for &'a mut String {
63//!     fn sl_from_ref_mut(value: Value, vm: &'a mut SloshVm) -> VMResult<Self> {
64//!         (&value).sl_as_mut(vm)
65//!     }
66//! }
67//!
68//! impl<'a> SlAsMut<'a, String> for &Value {
69//!     fn sl_as_mut(&mut self, vm: &'a mut SloshVm) -> VMResult<&'a mut String> {
70//!         match self {
71//!             Value::String(h) => vm.get_string_mut(*h),
72//!             _ => Err(VMError::new_conversion(
73//!                 ErrorStrings::fix_me_mismatched_type(
74//!                     <&'static str>::from(ValueType::String),
75//!                     self.display_type(vm),
76//!                 ),
77//!             )),
78//!         }
79//!     }
80//! }
81//!
82//! ```

252/// Use mutable [`SloshVm`] to take a rust value and convert it to a [`BridgedType`].
253pub trait SlFrom<T>: Sized
254where
255    Self: BridgedType,
256{
257    /// Converts to this type from the input type.
258    fn sl_from(value: T, vm: &mut SloshVm) -> VMResult<Self>;
259}
260
261impl<T> SlFrom<Vec<T>> for Value
262where
263    T: SlInto<Value>,
264{
265    fn sl_from(value: Vec<T>, vm: &mut SloshVm) -> VMResult<Self> {
266        let mut u = Vec::with_capacity(value.len());
267        for v in value {
268            u.push(v.sl_into(vm)?);
269        }
270        Ok(vm.alloc_vector(u))
271    }
272}
273
274impl<'a, T> SlFromRef<'a, slvm::Value> for Vec<T>
275where
276    T: SlFromRef<'a, Value> + 'a,
277{
278    fn sl_from_ref(value: slvm::Value, vm: &'a SloshVm) -> VMResult<Self> {
279        let mut res = vec![];
280        for val in value.iter(vm) {
281            let t: T = val.sl_into_ref(vm)?;
282            res.push(t);
283        }
284        Ok(res)
285    }
286}

288/// Inverse of [`SlFrom`]
289pub trait SlInto<T>: Sized
290where
291    T: BridgedType,
292{
293    /// Converts this type into the (usually inferred) input type.
294    fn sl_into(self, vm: &mut SloshVm) -> VMResult<T>;
295}
296
297impl<T, U> SlInto<U> for T
298where
299    U: SlFrom<T>,
300{
301    fn sl_into(self, vm: &mut SloshVm) -> VMResult<U> {
302        U::sl_from(self, vm)
303    }
304}

364/// Converts a [`BridgedType`] to some rust type
365pub trait SlAsRef<'a, T: ?Sized>
366where
367    Self: BridgedType,
368{
369    /// Converts this type into a shared reference of the (usually inferred) input type.
370    fn sl_as_ref(&self, vm: &'a SloshVm) -> VMResult<&'a T>;
371}
372
373// SlAsRef lifts over &
374impl<'a, T: ?Sized, U: ?Sized> SlAsRef<'a, U> for &'a T
375where
376    T: SlAsRef<'a, U>,
377    &'a T: BridgedType,
378{
379    #[inline]
380    fn sl_as_ref(&self, vm: &'a SloshVm) -> VMResult<&'a U> {
381        <T as SlAsRef<'a, U>>::sl_as_ref(*self, vm)
382    }
383}
