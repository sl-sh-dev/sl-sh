5fn get_global_with_ns(env: &mut SloshVm, si: Interned) -> u32 {
6    if env.env().get_namespace().name().is_empty() {
7        env.get_reserve_global(si)
8    } else {
9        let mut ns = env.env().get_namespace().name().to_string();
10        ns.push_str("::");
11        ns.push_str(env.get_interned(si));
12        let i = env.intern(&ns);
13        env.get_reserve_global(i)
14    }
15}

17pub(crate) fn compile_def(
18    env: &mut SloshVm,
19    state: &mut CompileState,
20    cdr: &[Value],
21    result: usize,
22) -> VMResult<()> {
23    match (cdr.len(), cdr.first()) {
24        (_, None) => return Err(VMError::new_compile("def: expected symbol")),
25        (1, Some(Value::Symbol(si))) => {
26            // 'def symbol' predeclares a symbol to be used later, no bytecode.
27            let si_const = get_global_with_ns(env, *si);
28            if let Some(doc_string) = state.doc_string {
29                let key = env.intern("doc-string");
30                env.set_global_property(si_const, key, doc_string);
31                state.doc_string = None;
32            }
33        }
34        (2, Some(Value::Symbol(si))) => {
35            let si_const = get_global_with_ns(env, *si);
36            if let Some(doc_string) = state.doc_string {
37                let key = env.intern("doc-string");
38                env.set_global_property(si_const, key, doc_string);
39                state.doc_string = None;
40            }
41            compile(env, state, cdr[1], result)?;
42            state
43                .chunk
44                .encode_def(result as u16, si_const, env.own_line(), false)?;
45        }
46        _ => return Err(VMError::new_compile("def: malformed")),
47    }
48    Ok(())
49}

51pub(crate) fn compile_set(
52    env: &mut SloshVm,
53    state: &mut CompileState,
54    cdr: &[Value],
55    result: usize,
56) -> VMResult<()> {
57    if cdr.len() == 2 {
58        if let Value::Symbol(si) = cdr[0] {
59            if let Some(idx) = state.get_symbol(si) {
60                compile(env, state, cdr[1], result)?;
61                state
62                    .chunk
63                    .encode2(SET, idx as u16, result as u16, env.own_line())?;
64            } else if let Some(si_const) = env.global_intern_slot(si) {
65                compile(env, state, cdr[1], result)?;
66                state
67                    .chunk
68                    .encode_def(result as u16, si_const, env.own_line(), false)?;
69            } else {
70                let sym = env.get_interned(si);
71                return Err(VMError::new_compile(format!(
72                    "Symbol {sym} not defined (maybe you need to use 'def {sym}' to pre-declare it)."
73                )));
74            }
75        } else {
76            match cdr[0].get_pair(env) {
77                Some((Value::Symbol(i), ncdr)) if i == env.specials().get => {
78                    if cdr.len() != 2 {
79                        return Err(VMError::new_compile(
80                            "Wrong number of arguments, expected two.",
81                        ));
82                    }
83                    compile(env, state, cdr[1], result)?;
84                    let mut cdr_i = ncdr.iter(env);
85                    let collection = cdr_i.next().expect("had two elements!");
86                    let field = cdr_i.next().expect("had two elements!");
87                    drop(cdr_i);
88                    compile(env, state, collection, result + 1)?;
89                    compile(env, state, field, result + 2)?;
90                    state.chunk.encode3(
91                        SETCOL,
92                        result as u16,
93                        (result + 1) as u16,
94                        (result + 2) as u16,
95                        env.own_line(),
96                    )?;
97                }
98                _ => return Err(VMError::new_compile("set!: expected symbol")),
99            }
100        }
101    } else {
102        return Err(VMError::new_compile("set!: malformed"));
103    }
104    Ok(())
105}
