9pub fn mk_state(
10    env: &mut SloshVm,
11    state: &mut CompileState,
12    args: Value,
13) -> VMResult<(CompileState, Vec<Value>, Vec<DestructType>)> {
14    let line = env.own_line().unwrap_or(1);
15    let mut new_state =
16        CompileState::new_state(state.chunk.file_name, line, Some(state.symbols.clone()));
17    let args_iter: Vec<Value> = get_args_iter(env, args, "fn")?.collect();
18    let mut opt = false;
19    let mut rest = false;
20    let mut opt_comps = Vec::new();
21    let mut destructures = Vec::new();
22    let mut next_is_opt = false;
23    new_state.chunk.dbg_args = Some(Vec::new());
24    let mut total_args = 0_usize;
25    for a in args_iter {
26        if next_is_opt {
27            opt_comps.pop();
28            opt_comps.push(a);
29            next_is_opt = false;
30            continue;
31        }
32        let a = resolve_destruct_containers(env, a);
33        match a {
34            Value::Symbol(i) => {
35                if i == env.specials().rest {
36                    rest = true;
37                } else if i == env.specials().optional {
38                    opt = true;
39                } else {
40                    new_state.symbols.borrow_mut().insert(i);
41                    if let Some(dbg_args) = new_state.chunk.dbg_args.as_mut() {
42                        dbg_args.push(i);
43                    }
44                    if opt {
45                        new_state.chunk.opt_args += 1;
46                        opt_comps.push(Value::Nil);
47                    } else {
48                        new_state.chunk.args += 1;
49                    }
50                    total_args += 1;
51                }
52            }
53            Value::Keyword(i) if i == env.specials().equal => {
54                if !opt {
55                    return Err(VMError::new_compile(
56                        "invalid args, := must come after % (optional)",
57                    ));
58                }
59                if next_is_opt {
60                    return Err(VMError::new_compile("invalid args, := := invalid"));
61                }
62                next_is_opt = true;
63            }
64            Value::Vector(handle) => {
65                new_state.symbols.borrow_mut().reserve_reg();
66                if let Some(dbg_args) = new_state.chunk.dbg_args.as_mut() {
67                    dbg_args.push(env.specials().scratch);
68                }
69                if opt {
70                    new_state.chunk.opt_args += 1;
71                    opt_comps.push(Value::Nil);
72                } else {
73                    new_state.chunk.args += 1;
74                }
75                total_args += 1;
76                destructures.push(DestructType::Vector(handle, total_args));
77            }
78            Value::Map(handle) => {
79                new_state.symbols.borrow_mut().reserve_reg();
80                if let Some(dbg_args) = new_state.chunk.dbg_args.as_mut() {
81                    dbg_args.push(env.specials().scratch);
82                }
83                if opt {
84                    new_state.chunk.opt_args += 1;
85                    opt_comps.push(Value::Nil);
86                } else {
87                    new_state.chunk.args += 1;
88                }
89                total_args += 1;
90                destructures.push(DestructType::Map(handle, total_args));
91            }
92            _ => {
93                return Err(VMError::new_compile(format!(
94                    "invalid args, must be symbols got {}/{}",
95                    a.display_type(env),
96                    a.display_value(env)
97                )));
98            }
99        }
100    }
101    new_state.chunk.rest = rest;
102    Ok((new_state, opt_comps, destructures))
103}

105pub(crate) fn compile_fn(
106    env: &mut SloshVm,
107    state: &mut CompileState,
108    args: Value,
109    cdr: &[Value],
110    result: usize,
111    is_macro: bool,
112) -> VMResult<()> {
113    let (mut new_state, opt_comps, destructure_patterns) = mk_state(env, state, args)?;
114    for r in cdr.iter() {
115        pass1(env, &mut new_state, *r)?;
116    }
117    let reserved = new_state.reserved_regs();
118    for (i, r) in opt_comps.into_iter().enumerate() {
119        let target_reg = new_state.chunk.args as usize + i + 1;
120        let jmp_idx = new_state.chunk.add_jump(0);
121        new_state
122            .chunk
123            .encode2(JMPNU, target_reg as u16, jmp_idx as u16, env.own_line())?;
124        compile(env, &mut new_state, r, reserved)?;
125        new_state
126            .chunk
127            .encode2(MOV, target_reg as u16, reserved as u16, env.own_line())?;
128        new_state
129            .chunk
130            .update_jump(jmp_idx, new_state.chunk.code.len() as u32);
131    }
132    let mut destruct_state = DestructState::new();
133    for destruct_type in destructure_patterns {
134        destruct_state.do_destructure(env, &mut new_state, destruct_type)?;
135        let mut free_reg = new_state.reserved_regs();
136        destruct_state.compile(env, &mut new_state, &mut free_reg)?;
137    }
138    let reserved = new_state.reserved_regs();
139    let last_thing = cdr.len() - 1;
140    for (i, r) in cdr.iter().enumerate() {
141        if i == last_thing {
142            new_state.tail = true;
143        }
144        compile(env, &mut new_state, *r, reserved)?;
145    }
146    new_state
147        .chunk
148        .encode1(SRET, reserved as u16, env.own_line())?;
149    let mut closure = false;
150    if !new_state.symbols.borrow().captures.borrow().is_empty() {
151        let mut caps = Vec::new();
152        for (_, _, c) in new_state.symbols.borrow().captures.borrow().iter() {
153            caps.push(*c as u32);
154        }
155        new_state.chunk.captures = Some(caps);
156        closure = true;
157    }
158    new_state.chunk.input_regs = reserved;
159    new_state.chunk.extra_regs = new_state.max_regs - reserved;
160    env.pause_gc();
161    let lambda = env.alloc_lambda(Arc::new(new_state.chunk));
162    env.unpause_gc();
163    if is_macro {
164        // Unwrap safe since we just allocated lambda on the heap.
165        env.set_heap_property(lambda, ":macro", Value::True);
166    }
167    let const_i = state.add_constant(lambda);
168    state
169        .chunk
170        .encode2(CONST, result as u16, const_i as u16, env.own_line())?;
171    if closure {
172        state
173            .chunk
174            .encode2(CLOSE, result as u16, result as u16, env.own_line())?;
175    }
176    Ok(())
177}
