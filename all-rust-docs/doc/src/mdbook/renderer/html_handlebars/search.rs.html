19/// Tokenizes in the same way as elasticlunr-rs (for English), but also drops long tokens.
20fn tokenize(text: &str) -> Vec<String> {
21    text.split(|c: char| c.is_whitespace() || c == '-')
22        .filter(|s| !s.is_empty())
23        .map(|s| s.trim().to_lowercase())
24        .filter(|s| s.len() <= MAX_WORD_LENGTH_TO_INDEX)
25        .collect()
26}

28/// Creates all files required for search.
29pub fn create_files(search_config: &Search, destination: &Path, book: &Book) -> Result<()> {
30    let mut index = IndexBuilder::new()
31        .add_field_with_tokenizer("title", Box::new(&tokenize))
32        .add_field_with_tokenizer("body", Box::new(&tokenize))
33        .add_field_with_tokenizer("breadcrumbs", Box::new(&tokenize))
34        .build();
35
36    let mut doc_urls = Vec::with_capacity(book.sections.len());
37
38    let chapter_configs = sort_search_config(&search_config.chapter);
39    validate_chapter_config(&chapter_configs, book)?;
40
41    for item in book.iter() {
42        let chapter = match item {
43            BookItem::Chapter(ch) if !ch.is_draft_chapter() => ch,
44            _ => continue,
45        };
46        let chapter_settings =
47            get_chapter_settings(&chapter_configs, chapter.source_path.as_ref().unwrap());
48        if !chapter_settings.enable.unwrap_or(true) {
49            continue;
50        }
51        render_item(&mut index, search_config, &mut doc_urls, chapter)?;
52    }
53
54    let index = write_to_json(index, search_config, doc_urls)?;
55    debug!("Writing search index âœ“");
56    if index.len() > 10_000_000 {
57        warn!("searchindex.json is very large ({} bytes)", index.len());
58    }
59
60    if search_config.copy_js {
61        utils::fs::write_file(destination, "searchindex.json", index.as_bytes())?;
62        utils::fs::write_file(
63            destination,
64            "searchindex.js",
65            format!("Object.assign(window.search, {index});").as_bytes(),
66        )?;
67        utils::fs::write_file(destination, "searcher.js", searcher::JS)?;
68        utils::fs::write_file(destination, "mark.min.js", searcher::MARK_JS)?;
69        utils::fs::write_file(destination, "elasticlunr.min.js", searcher::ELASTICLUNR_JS)?;
70        debug!("Copying search files âœ“");
71    }
72
73    Ok(())
74}

76/// Uses the given arguments to construct a search document, then inserts it to the given index.
77fn add_doc(
78    index: &mut Index,
79    doc_urls: &mut Vec<String>,
80    anchor_base: &str,
81    heading: &str,
82    id_counter: &mut HashMap<String, usize>,
83    section_id: &Option<CowStr<'_>>,
84    items: &[&str],
85) {
86    // Either use the explicit section id the user specified, or generate one
87    // from the heading content.
88    let section_id = section_id.as_ref().map(|id| id.to_string()).or_else(|| {
89        if heading.is_empty() {
90            // In the case where a chapter has no heading, don't set a section id.
91            None
92        } else {
93            Some(utils::unique_id_from_content(heading, id_counter))
94        }
95    });
96
97    let url = if let Some(id) = section_id {
98        Cow::Owned(format!("{anchor_base}#{id}"))
99    } else {
100        Cow::Borrowed(anchor_base)
101    };
102    let url = utils::collapse_whitespace(url.trim());
103    let doc_ref = doc_urls.len().to_string();
104    doc_urls.push(url.into());
105
106    let items = items.iter().map(|&x| utils::collapse_whitespace(x.trim()));
107    index.add_doc(&doc_ref, items);
108}

110/// Renders markdown into flat unformatted text and adds it to the search index.
111fn render_item(
112    index: &mut Index,
113    search_config: &Search,
114    doc_urls: &mut Vec<String>,
115    chapter: &Chapter,
116) -> Result<()> {
117    let chapter_path = chapter
118        .path
119        .as_ref()
120        .expect("Checked that path exists above");
121    let filepath = Path::new(&chapter_path).with_extension("html");
122    let filepath = filepath
123        .to_str()
124        .with_context(|| "Could not convert HTML path to str")?;
125    let anchor_base = utils::fs::normalize_path(filepath);
126
127    let mut p = utils::new_cmark_parser(&chapter.content, false).peekable();
128
129    let mut in_heading = false;
130    let max_section_depth = u32::from(search_config.heading_split_level);
131    let mut section_id = None;
132    let mut heading = String::new();
133    let mut body = String::new();
134    let mut breadcrumbs = chapter.parent_names.clone();
135    let mut footnote_numbers = HashMap::new();
136
137    breadcrumbs.push(chapter.name.clone());
138
139    let mut id_counter = HashMap::new();
140    while let Some(event) = p.next() {
141        match event {
142            Event::Start(Tag::Heading { level, id, .. }) if level as u32 <= max_section_depth => {
143                if !heading.is_empty() {
144                    // Section finished, the next heading is following now
145                    // Write the data to the index, and clear it for the next section
146                    add_doc(
147                        index,
148                        doc_urls,
149                        &anchor_base,
150                        &heading,
151                        &mut id_counter,
152                        &section_id,
153                        &[&heading, &body, &breadcrumbs.join(" Â» ")],
154                    );
155                    heading.clear();
156                    body.clear();
157                    breadcrumbs.pop();
158                }
159
160                section_id = id;
161                in_heading = true;
162            }
163            Event::End(TagEnd::Heading(level)) if level as u32 <= max_section_depth => {
164                in_heading = false;
165                breadcrumbs.push(heading.clone());
166            }
167            Event::Start(Tag::FootnoteDefinition(name)) => {
168                let number = footnote_numbers.len() + 1;
169                footnote_numbers.entry(name).or_insert(number);
170            }
171            Event::Html(html) => {
172                let mut html_block = html.into_string();
173
174                // As of pulldown_cmark 0.6, html events are no longer contained
175                // in an HtmlBlock tag. We must collect consecutive Html events
176                // into a block ourselves.
177                while let Some(Event::Html(html)) = p.peek() {
178                    html_block.push_str(html);
179                    p.next();
180                }
181                body.push_str(&clean_html(&html_block));
182            }
183            Event::InlineHtml(html) => {
184                // This is not capable of cleaning inline tags like
185                // `foo <script>â€¦</script>`. The `<script>` tags show up as
186                // individual InlineHtml events, and the content inside is
187                // just a regular Text event. There isn't a very good way to
188                // know how to collect all the content in-between. I'm not
189                // sure if this is easily fixable. It should be extremely
190                // rare, since script and style tags should almost always be
191                // blocks, and worse case you have some noise in the index.
192                body.push_str(&clean_html(&html));
193            }
194            Event::Start(_) | Event::End(_) | Event::Rule | Event::SoftBreak | Event::HardBreak => {
195                // Insert spaces where HTML output would usually separate text
196                // to ensure words don't get merged together
197                if in_heading {
198                    heading.push(' ');
199                } else {
200                    body.push(' ');
201                }
202            }
203            Event::Text(text) | Event::Code(text) => {
204                if in_heading {
205                    heading.push_str(&text);
206                } else {
207                    body.push_str(&text);
208                }
209            }
210            Event::FootnoteReference(name) => {
211                let len = footnote_numbers.len() + 1;
212                let number = footnote_numbers.entry(name).or_insert(len);
213                body.push_str(&format!(" [{number}] "));
214            }
215            Event::TaskListMarker(_checked) => {}
216        }
217    }
218
219    if !body.is_empty() || !heading.is_empty() {
220        let title = if heading.is_empty() {
221            if let Some(chapter) = breadcrumbs.first() {
222                chapter
223            } else {
224                ""
225            }
226        } else {
227            &heading
228        };
229        // Make sure the last section is added to the index
230        add_doc(
231            index,
232            doc_urls,
233            &anchor_base,
234            &heading,
235            &mut id_counter,
236            &section_id,
237            &[title, &body, &breadcrumbs.join(" Â» ")],
238        );
239    }
240
241    Ok(())
242}

244fn write_to_json(index: Index, search_config: &Search, doc_urls: Vec<String>) -> Result<String> {
245    use elasticlunr::config::{SearchBool, SearchOptions, SearchOptionsField};
246    use std::collections::BTreeMap;
247
248    #[derive(Serialize)]
249    struct ResultsOptions {
250        limit_results: u32,
251        teaser_word_count: u32,
252    }
253
254    #[derive(Serialize)]
255    struct SearchindexJson {
256        /// The options used for displaying search results
257        results_options: ResultsOptions,
258        /// The searchoptions for elasticlunr.js
259        search_options: SearchOptions,
260        /// Used to lookup a document's URL from an integer document ref.
261        doc_urls: Vec<String>,
262        /// The index for elasticlunr.js
263        index: elasticlunr::Index,
264    }
265
266    let mut fields = BTreeMap::new();
267    let mut opt = SearchOptionsField::default();
268    let mut insert_boost = |key: &str, boost| {
269        opt.boost = Some(boost);
270        fields.insert(key.into(), opt);
271    };
272    insert_boost("title", search_config.boost_title);
273    insert_boost("body", search_config.boost_paragraph);
274    insert_boost("breadcrumbs", search_config.boost_hierarchy);
275
276    let search_options = SearchOptions {
277        bool: if search_config.use_boolean_and {
278            SearchBool::And
279        } else {
280            SearchBool::Or
281        },
282        expand: search_config.expand,
283        fields,
284    };
285
286    let results_options = ResultsOptions {
287        limit_results: search_config.limit_results,
288        teaser_word_count: search_config.teaser_word_count,
289    };
290
291    let json_contents = SearchindexJson {
292        results_options,
293        search_options,
294        doc_urls,
295        index,
296    };
297
298    // By converting to serde_json::Value as an intermediary, we use a
299    // BTreeMap internally and can force a stable ordering of map keys.
300    let json_contents = serde_json::to_value(&json_contents)?;
301    let json_contents = serde_json::to_string(&json_contents)?;
302
303    Ok(json_contents)
304}

306fn clean_html(html: &str) -> String {
307    static AMMONIA: Lazy<ammonia::Builder<'static>> = Lazy::new(|| {
308        let mut clean_content = HashSet::new();
309        clean_content.insert("script");
310        clean_content.insert("style");
311        let mut builder = ammonia::Builder::new();
312        builder
313            .tags(HashSet::new())
314            .tag_attributes(HashMap::new())
315            .generic_attributes(HashSet::new())
316            .link_rel(None)
317            .allowed_classes(HashMap::new())
318            .clean_content_tags(clean_content);
319        builder
320    });
321    AMMONIA.clean(html).to_string()
322}

324fn validate_chapter_config(
325    chapter_configs: &[(PathBuf, SearchChapterSettings)],
326    book: &Book,
327) -> Result<()> {
328    for (path, _) in chapter_configs {
329        let found = book
330            .iter()
331            .filter_map(|item| match item {
332                BookItem::Chapter(ch) if !ch.is_draft_chapter() => Some(ch),
333                _ => None,
334            })
335            .any(|chapter| {
336                let ch_path = chapter.source_path.as_ref().unwrap();
337                ch_path.starts_with(path)
338            });
339        if !found {
340            bail!(
341                "[output.html.search.chapter] key `{}` does not match any chapter paths",
342                path.display()
343            );
344        }
345    }
346    Ok(())
347}

349fn sort_search_config(
350    map: &HashMap<String, SearchChapterSettings>,
351) -> Vec<(PathBuf, SearchChapterSettings)> {
352    let mut settings: Vec<_> = map
353        .iter()
354        .map(|(key, value)| (PathBuf::from(key), value.clone()))
355        .collect();
356    // Note: This is case-sensitive, and assumes the author uses the same case
357    // as the actual filename.
358    settings.sort_by(|a, b| a.0.cmp(&b.0));
359    settings
360}

362fn get_chapter_settings(
363    chapter_configs: &[(PathBuf, SearchChapterSettings)],
364    source_path: &Path,
365) -> SearchChapterSettings {
366    let mut result = SearchChapterSettings::default();
367    for (path, config) in chapter_configs {
368        if source_path.starts_with(path) {
369            result.enable = config.enable.or(result.enable);
370        }
371    }
372    result
373}
