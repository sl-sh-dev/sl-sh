8/// pr/prn/dasm/dump-globals which write directly to stdout are handed new behavior according
9/// to variable: [`NoopSwap`].
10pub fn do_to_vm_stdout_fcns(env: &mut SloshVm, noop_swap_type: NoopSwap) {
11    let _ = noop_swap_internal(env, "pr".to_string(), noop_swap_type);
12    let _ = noop_swap_internal(env, "prn".to_string(), noop_swap_type);
13    let _ = noop_swap_internal(env, "dasm".to_string(), noop_swap_type);
14    let _ = noop_swap_internal(env, "dump-globals".to_string(), noop_swap_type);
15}
16
17/// pr/prn/dasm/dump-globals which write directly to stdout are mapped to noop
18pub fn vm_with_stdout_disabled(env: &mut SloshVm) {
19    do_to_vm_stdout_fcns(env, NoopSwap::MakeNoop);
20}
21
22/// pr/prn/dasm/dump-globals which write directly to stdout are not mapped to noop
23pub fn vm_with_stdout_enabled(env: &mut SloshVm) {
24    do_to_vm_stdout_fcns(env, NoopSwap::MakeNotNoop);
25}

27/// If noop_stdout is set to true then all functions that write to stdout
28/// (pr/prn/dasm/dump-globals) will be overwritten with the noop function.
29pub fn vm_with_builtins_and_core(env: &mut SloshVm, noop_stdout: bool) -> VMResult<()> {
30    docs::add_builtins(env);
31    slosh_lib::set_builtins(env);
32    if noop_stdout {
33        // must be called at this point. before set_shell_builtins. because
34        // set_shell_builtins calls set_environment which loads the sloshrc
35        // which is allowed to write to stdout and is something that should *NOT*
36        // happen while this VM is being used.
37        vm_with_stdout_disabled(env);
38    }
39    slosh_lib::set_shell_builtins(env);
40    slosh_lib::load_builtins_lisp_less_sloshrc(env)
41}

43pub fn add_user_builtins(env: &mut SloshVm, load_paths: &[String], files_to_load: &[String]) {
44    let _ = slosh_lib::load_builtins_lisp_less_sloshrc(env);
45
46    let load_paths: Vec<&str> = load_paths.iter().map(AsRef::as_ref).collect();
47    slosh_lib::set_initial_load_path(env, load_paths);
48    for script in files_to_load {
49        let script = env.intern(script);
50        let script = env.get_interned(script);
51        let _ = load_eval::load_internal(env, script);
52    }
53}
