113enum Namespace {
114    Global,
115    Other(Interned),
116}

118impl Namespace {
119    fn display(&self, vm: &mut SloshVm) -> String {
120        match self {
121            Namespace::Global => GLOBAL_NAMESPACE.to_string(),
122            Namespace::Other(s) => {
123                let s = vm.get_interned(*s);
124                s.to_string()
125            }
126        }
127    }
128
129    fn get_doc(
130        &self,
131        interned: &Interned,
132        docs: &mut Vec<SloshDoc>,
133        vm: &mut SloshVm,
134        require_proper_format: bool,
135    ) -> DocResult<()> {
136        let slosh_doc = SloshDoc::new(*interned, vm, self.clone());
137        match slosh_doc {
138            Ok(slosh_doc) => {
139                docs.push(slosh_doc);
140            }
141            Err(e) => match e {
142                _ if !require_proper_format => {
143                    let incomplete_doc = SloshDoc::new_incomplete(*interned, vm, self.clone())?;
144                    docs.push(incomplete_doc);
145                }
146                _ => {
147                    return Err(e);
148                }
149            },
150        }
151        Ok(())
152    }
153
154    fn add_docs(
155        &self,
156        docs: &mut Vec<SloshDoc>,
157        vm: &mut SloshVm,
158        require_proper_format: bool,
159    ) -> DocResult<()> {
160        match self {
161            Namespace::Global => {
162                for g in vm.globals().clone().keys() {
163                    self.get_doc(g, docs, vm, require_proper_format)?;
164                }
165            }
166            Namespace::Other(i) => {
167                let value = builtins::retrieve_in_namespace(vm, i);
168                for v in value {
169                    if let Value::Symbol(sym) = v {
170                        self.get_doc(&sym, docs, vm, require_proper_format)?;
171                    }
172                }
173            }
174        }
175        docs.sort();
176        docs.dedup();
177        Ok(())
178    }
179}

212impl DocStringSection {
213    pub fn from_symbol(slot: u32, sym: Value, vm: &mut SloshVm) -> DocResult<DocStringSection> {
214        let sym_str = sym.display_value(vm);
215        let raw_doc_string = Self::raw_docstring(slot, vm);
216        let backup_usage = slosh_lib::usage(vm, slot, &sym);
217        Self::parse_doc_string(Cow::Owned(sym_str), raw_doc_string, backup_usage)
218    }

220    fn raw_docstring(slot: u32, vm: &mut SloshVm) -> String {
221        let docstring_key = vm.intern_static("doc-string");
222        vm.get_global_property(slot, docstring_key)
223            .and_then(|x| match x {
224                Value::String(h) => Some(vm.get_string(h).to_string()),
225                Value::StringConst(i) => Some(vm.get_interned(i).to_string()),
226                _ => None,
227            })
228            // return default empty string and have parse_doc_string handle error if no doc provided.
229            .unwrap_or_default()
230    }

232    /// Given the rules for parsing slosh docstrings, parse one! See [`static@DOC_REGEX`]
233    /// for the specification.
234    pub fn parse_doc_string(
235        symbol: Cow<'_, str>,
236        raw_doc_string: String,
237        backup_usage: String,
238    ) -> DocResult<DocStringSection> {
239        let cap =
240            DOC_REGEX
241                .captures(raw_doc_string.as_str())
242                .ok_or_else(|| DocError::NoDocString {
243                    symbol: symbol.to_string(),
244                });
245        if EXEMPTIONS.contains(symbol.as_ref()) && cap.is_err() {
246            let usage = Some("unknown".to_string());
247            let description = "unknown".to_string();
248            let section = "undocumented".to_string();
249            let example = None;
250            return Ok(DocStringSection {
251                usage,
252                description,
253                section,
254                example,
255            });
256        }
257        let cap = cap?;
258        let mut usage = cap.get(2).map(|x| x.as_str().trim().to_string());
259        if usage.is_none() && !backup_usage.trim().is_empty() {
260            usage = Some(backup_usage);
261        }
262        let description = cap
263            .get(3)
264            .ok_or_else(|| DocError::DocStringMissingSection {
265                symbol: symbol.to_string(),
266                section: "Description".to_string(),
267            })
268            .map(|x| x.as_str().to_string())?;
269        let section = cap
270            .get(4)
271            .ok_or_else(|| DocError::DocStringMissingSection {
272                symbol: symbol.to_string(),
273                section: "Section".to_string(),
274            })
275            .map(|x| x.as_str().trim().to_string())?;
276        let example = cap.get(6).map(|x| x.as_str().trim().to_string());
277        if EXEMPTIONS.contains(symbol.as_ref()) {
278            Err(DocError::RemoveExemption {
279                symbol: symbol.to_string(),
280            })
281        } else {
282            Ok(DocStringSection {
283                usage,
284                description,
285                section,
286                example,
287            })
288        }
289    }

475enum DocError {
476    NoSymbol { symbol: String },
477    NoDocString { symbol: String },
478    RemoveExemption { symbol: String },
479    DocStringMissingSection { symbol: String, section: String },
480}

484        Display::fmt(self, f)
485    }

488impl Display for DocError {
489    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
490        let str = match self {
491            DocError::NoSymbol { symbol } => {
492                format!(
493                    "No documentation string provided for symbol {symbol}."
494                )
495            }
496            DocError::NoDocString { symbol } => {
497                format!(
498                    "Either documentation provided does not conform to conventional layout or no documentation string provided for symbol {symbol} all slosh functions with documentation must have a string that conforms to the conventional layout."
499                )
500            }
501            DocError::DocStringMissingSection { symbol, section } => {
502                format!("Invalid documentation string for symbol {symbol}, missing required section {section:?}")
503            }
504            DocError::RemoveExemption { symbol} => {
505                format!("Documentation has been added for {symbol}, remove it from EXEMPTIONS list in slosh_test::docs::EXEMPTIONS.")
506            }
507        }
508            .to_string();
509        write!(f, "{}", str)
510    }

566                Value::StringConst(i) => options.default_color = vm.get_interned(i).to_string(),

579fn highlight_matches(text: &str, query: &DocSearchQuery, style: &StyleOptions) -> String {
580    if query.use_regex {
581        highlight_regex_matches(text, &query.query, style)
582    } else if query.basic {
583        highlight_basic_matches(text, &query.query, style)
584    } else {
585        highlight_substring_matches(text, &query.query, style)
586    }
587}

589fn highlight_substring_matches(text: &str, query: &str, style: &StyleOptions) -> String {
590    let text_lower = text.to_lowercase();
591    let query_lower = query.to_lowercase();
592    let mut result = String::new();
593    let mut last_end = 0;
594
595    for (start, _) in text_lower.match_indices(&query_lower) {
596        // Add text before match
597        result.push_str(&text[last_end..start]);
598        // Add highlighted match
599        result.push_str(&style.highlight_color);
600        result.push_str(&text[start..start + query.len()]);
601        result.push_str(&style.default_color);
602        last_end = start + query.len();
603    }
604
605    // Add remaining text
606    result.push_str(&text[last_end..]);
607    result
608}

610fn highlight_regex_matches(text: &str, pattern: &str, style: &StyleOptions) -> String {
611    if let Ok(re) = Regex::new(pattern) {
612        let mut result = String::new();
613        let mut last_end = 0;
614
615        for mat in re.find_iter(text) {
616            // Add text before match
617            result.push_str(&text[last_end..mat.start()]);
618            // Add highlighted match
619            result.push_str(&style.highlight_color);
620            result.push_str(mat.as_str());
621            result.push_str(&style.default_color);
622            last_end = mat.end();
623        }
624
625        // Add remaining text
626        result.push_str(&text[last_end..]);
627        result
628    } else {
629        text.to_string()
630    }
631}

633fn highlight_basic_matches(text: &str, pattern: &str, style: &StyleOptions) -> String {
634    let pattern_lower = pattern.to_lowercase();
635    let mut result = String::new();
636    let mut pattern_chars = pattern_lower.chars().peekable();
637    let mut in_match = false;
638
639    for ch in text.chars() {
640        if let Some(&pattern_ch) = pattern_chars.peek() {
641            if ch.to_lowercase().next() == Some(pattern_ch) {
642                if !in_match {
643                    result.push_str(&style.highlight_color);
644                    in_match = true;
645                }
646                result.push(ch);
647                pattern_chars.next();
648            } else {
649                if in_match {
650                    result.push_str(&style.default_color);
651                    in_match = false;
652                }
653                result.push(ch);
654            }
655        } else {
656            if in_match {
657                result.push_str(&style.default_color);
658                in_match = false;
659            }
660            result.push(ch);
661        }
662    }
663
664    if in_match {
665        result.push_str(&style.default_color);
666    }
667
668    result
669}

671fn generate_styled_output(
672    _vm: &mut SloshVm,
673    results: &BTreeSet<SloshDoc>,
674    query: &DocSearchQuery,
675    style: &StyleOptions,
676) -> String {
677    let mut output = String::new();
678    let len = results.len();
679
680    for (i, doc) in results.iter().enumerate() {
681        if i > 0 {
682            output.push_str("\n────────────────────────────────────────\n\n");
683        }
684
685        // Result header with highlighted symbol
686        output.push_str(&format!("Result {}: ", i + 1));
687        let mut doc = doc.clone();
688        doc = doc.into_styled_output(query, style);
689        output.push_str(&format!("{doc}"));
690    }
691
692    output.push_str(&format!(
693        "\nFound {} result{}.",
694        len,
695        if len == 1 { "" } else { "s" }
696    ));
697
698    output

766/// Each doc has a tag in its `Section:` definition by convention that logically groups functions.
767/// Using a HashMap store the section tags as keys and add all slosh docs from to a vector as a value
768/// corresponding to its section.
769fn get_docs_by_section(
770    vm: &mut SloshVm,
771    require_proper_format: bool,
772) -> HashMap<String, BTreeSet<SloshDoc>> {
773    let mut docs_by_section: HashMap<String, BTreeSet<SloshDoc>> = HashMap::new();
774    let mut docs: Vec<SloshDoc> = vec![];
775    Namespace::Global
776        .add_docs(&mut docs, vm, require_proper_format)
777        .unwrap();
778    let namespaces = builtins::get_namespaces_interned(vm);
779    for i in namespaces {
780        let namespace = Namespace::Other(i);
781        namespace
782            .add_docs(&mut docs, vm, require_proper_format)
783            .unwrap();
784    }
785    for d in docs {
786        let d = d.clone();
787        let section = d.doc_string.section.clone();
788        docs_by_section.entry(section).or_default().insert(d);
789    }
790    docs_by_section
791}

793fn build_symbols_list(
794    docs_by_section: &BTreeMap<String, BTreeSet<SloshDoc>>,
795    namer: fn(&str, &SloshDoc) -> String,
796) -> BTreeMap<String, String> {
797    let mut map = BTreeMap::new();
798    for (section, v) in docs_by_section.iter() {

800        let len = v.len();
801        for (i, docs) in v.iter().enumerate() {
802            let name = namer(section.as_ref(), docs);
803            list += &name;
804
805            if i + 1 != len {
806                list += ", ";
807            }
808        }
809        list += "\n";
810        map.insert(section.to_string(), list);
811    }
812    map
813}

1131impl DocSearchQuery {
1132    fn new(query: String) -> Self {

1134            query,
1135            use_regex: false,

1139            section_filter: None,
1140        }
1141    }

1144        self.use_regex = true;
1145        self.basic = false;
1146        self
1147    }

1150        self.basic = true;
1151        self.use_regex = false;
1152        self
1153    }

1156        self.fields = fields.into_iter().collect();
1157        self
1158    }

1161        self.namespace_filter = Some(namespace);

1166        self.section_filter = Some(section);

1168    }
1169
1170    fn should_search_field(&self, field: &str) -> bool {
1171        self.fields.is_empty() || self.fields.contains(field)
1172    }
1173
1174    fn matches_text(&self, text: &str) -> bool {
1175        if self.use_regex {
1176            if let Ok(re) = Regex::new(&self.query) {
1177                re.is_match(text)
1178            } else {
1179                false
1180            }
1181        } else if self.basic {
1182            basic_match(text, &self.query)
1183        } else {
1184            text.to_lowercase().contains(&self.query.to_lowercase())
1185        }
1186    }
1187}

1189fn basic_match(text: &str, pattern: &str) -> bool {
1190    let text_lower = text.to_lowercase();
1191    let pattern_lower = pattern.to_lowercase();
1192    let mut pattern_chars = pattern_lower.chars();
1193    let mut current_char = pattern_chars.next();
1194
1195    for text_char in text_lower.chars() {
1196        if let Some(p_char) = current_char {
1197            if text_char == p_char {
1198                current_char = pattern_chars.next();
1199            }
1200        } else {
1201            return true;
1202        }
1203    }
1204
1205    current_char.is_none()
1206}

1208fn search_docs(
1209    vm: &mut SloshVm,
1210    query: &DocSearchQuery,
1211    require_proper_format: bool,
1212) -> BTreeSet<SloshDoc> {
1213    let docs_by_section = get_docs_by_section(vm, require_proper_format);
1214    let mut results = BTreeSet::new();
1215
1216    for (section, docs) in docs_by_section {
1217        // Check section filter
1218        if let Some(ref section_filter) = query.section_filter {
1219            if !query.matches_text(&section) && section != *section_filter {
1220                continue;
1221            }
1222        }
1223
1224        for doc in docs {
1225            // Check namespace filter
1226            if let Some(ref ns_filter) = query.namespace_filter {
1227                if !doc.namespace.contains(ns_filter) && doc.namespace != *ns_filter {
1228                    continue;
1229                }
1230            }
1231
1232            let mut matched = false;
1233
1234            // Search in symbol name
1235            if query.should_search_field("name")
1236                || query.should_search_field("symbol") && query.matches_text(&doc.symbol)
1237            {
1238                matched = true;
1239            }
1240
1241            // Search in usage
1242            if !matched
1243                && query.should_search_field(USAGE)
1244                && let Some(ref usage) = doc.doc_string.usage
1245                && query.matches_text(usage)
1246            {
1247                matched = true;
1248            }
1249
1250            // Search in description
1251            if !matched
1252                && query.should_search_field(DESCRIPTION)
1253                && query.matches_text(&doc.doc_string.description)
1254            {
1255                matched = true;
1256            }
1257
1258            // Search in section
1259            if !matched
1260                && query.should_search_field(SECTION)
1261                && query.matches_text(&doc.doc_string.section)
1262            {
1263                matched = true;
1264            }
1265
1266            // Search in example
1267            if !matched
1268                && query.should_search_field(EXAMPLE)
1269                && let Some(ref example) = doc.doc_string.example
1270                && query.matches_text(example)
1271            {
1272                matched = true;
1273            }
1274
1275            // If no specific fields specified, search all fields
1276            if !matched && query.fields.is_empty() {
1277                matched = query.matches_text(&doc.symbol)
1278                    || doc
1279                        .doc_string
1280                        .usage
1281                        .as_ref()
1282                        .map(|u| query.matches_text(u))
1283                        .unwrap_or(false)
1284                    || query.matches_text(&doc.doc_string.description)
1285                    || query.matches_text(&doc.doc_string.section)
1286                    || doc
1287                        .doc_string
1288                        .example
1289                        .as_ref()
1290                        .map(|e| query.matches_text(e))
1291                        .unwrap_or(false);
1292            }
1293
1294            if matched {
1295                results.insert(doc);
1296            }
1297        }
1298    }
1299
1300    results
1301}

1304    if registers.is_empty() {
1305        return Err(VMError::new_vm(
1306            "doc-search requires at least a query string".to_string(),
1307        ));
1308    }
1309
1310    let query_str = match &registers[0] {
1311        Value::String(s) => vm.get_string(*s).to_string(),
1312        Value::StringConst(i) => vm.get_interned(*i).to_string(),
1313        _ => {
1314            return Err(VMError::new_vm(
1315                "First argument must be a query string".to_string(),
1316            ));
1317        }
1318    };
1319
1320    let mut query = DocSearchQuery::new(query_str);
1321
1322    // Parse options if provided as a hash map
1323    let mut use_styled = true;
1324    let mut style_options = StyleOptions::default();
1325    let mut style_map_handle_opt = None;
1326    if registers.len() > 1 {
1327        match &registers[1] {
1328            Value::Map(handle) => {
1329                // Intern all keys first to avoid borrowing issues
1330                let regex_key = Value::Keyword(vm.intern_static("regex"));
1331                let basic_search_key = Value::Keyword(vm.intern_static("basic"));
1332                let fields_key = Value::Keyword(vm.intern_static("fields"));
1333                let ns_key = Value::Keyword(vm.intern_static("namespace"));
1334                let section_key = Value::Keyword(vm.intern_static("section"));
1335                let styled_key = Value::Keyword(vm.intern_static("styled"));
1336                let style_key = Value::Keyword(vm.intern_static("style"));
1337
1338                let map = vm.get_map(*handle);
1339
1340                // Check for regex option
1341                if let Some(Value::True) = map.get(vm, regex_key) {
1342                    query = query.with_regex();
1343                }
1344
1345                // Check for basic option
1346                if let Some(Value::True) = map.get(vm, basic_search_key) {
1347                    query = query.with_basic();
1348                }
1349
1350                // Check for fields option
1351                if let Some(Value::Vector(v_handle)) = map.get(vm, fields_key) {
1352                    let vec = vm.get_vector(v_handle);
1353                    let mut fields = Vec::new();
1354                    for val in vec.iter() {
1355                        match val {
1356                            Value::String(s) => fields.push(vm.get_string(*s).to_string()),
1357                            Value::StringConst(i) => fields.push(vm.get_interned(*i).to_string()),
1358                            Value::Keyword(i) => fields.push(vm.get_interned(*i).to_string()),
1359                            _ => {}
1360                        }
1361                    }
1362                    query = query.with_fields(fields);
1363                }
1364
1365                // Check for namespace filter
1366                if let Some(ns_val) = map.get(vm, ns_key) {
1367                    match ns_val {
1368                        Value::String(s) => {
1369                            query = query.with_namespace(vm.get_string(s).to_string())
1370                        }
1371                        Value::StringConst(i) => {
1372                            query = query.with_namespace(vm.get_interned(i).to_string())
1373                        }
1374                        _ => {}
1375                    }
1376                }
1377
1378                // Check for section filter
1379                if let Some(section_val) = map.get(vm, section_key) {
1380                    match section_val {
1381                        Value::String(s) => {
1382                            query = query.with_section(vm.get_string(s).to_string())
1383                        }
1384                        Value::StringConst(i) => {
1385                            query = query.with_section(vm.get_interned(i).to_string())
1386                        }
1387                        _ => {}
1388                    }
1389                }
1390
1391                // Check for styled option
1392                if let Some(Value::False) = map.get(vm, styled_key) {
1393                    use_styled = false;
1394                }
1395
1396                // Check for style customization
1397                if let Some(Value::Map(style_handle)) = map.get(vm, style_key) {
1398                    use_styled = true; // Enable styling if style options are provided
1399                    style_map_handle_opt = Some(style_handle);
1400                }
1401            }
1402            _ => {
1403                return Err(VMError::new_vm(
1404                    "Second argument must be a map of options".to_string(),
1405                ));
1406            }
1407        }
1408    }
1409
1410    // Process style options after we're done with the initial map
1411    if let Some(style_handle) = style_map_handle_opt {
1412        style_options = StyleOptions::from_handle(vm, style_handle);
1413    }
1414
1415    // Check if markdown output is requested
1416    let mut use_markdown = false;
1417    if registers.len() > 1 {
1418        if let Value::Map(handle) = &registers[1] {
1419            let markdown_key = Value::Keyword(vm.intern_static("markdown"));
1420            let map = vm.get_map(*handle);
1421            if let Some(Value::True) = map.get(vm, markdown_key) {
1422                use_markdown = true;
1423            }
1424        }
1425    }
1426
1427    let results = search_docs(vm, &query, false);
1428
1429    let result = if use_markdown {
1430        // Generate markdown output
1431        let mut markdown_output = String::new();
1432        let len = results.len();
1433        for (i, doc) in results.iter().enumerate() {
1434            // Don't print to stdout when generating markdown - the numbered list is included in the markdown output instead
1435            if i > 0 {
1436                markdown_output.push_str("\n---\n\n");
1437            }
1438            // Add a header showing which result this is
1439            markdown_output.push_str(&format!(
1440                "**Result {}: `{}::{}`**\n\n",
1441                i + 1,
1442                doc.namespace,
1443                doc.symbol
1444            ));
1445            markdown_output.push_str(&doc.as_md());
1446        }
1447        markdown_output.push_str(&format!(
1448            "\nFound {len} result{}.",
1449            if len == 1 { "" } else { "s" }
1450        ));
1451        vm.alloc_string(markdown_output)
1452    } else if use_styled {
1453        // Generate styled string output
1454        let styled_output = generate_styled_output(vm, &results, &query, &style_options);
1455        vm.alloc_string(styled_output)
1456    } else {
1457        // Generate vector of doc maps
1458        let mut result_values = Vec::new();
1459        for doc in results {
1460            match Value::sl_from(doc, vm) {
1461                Ok(val) => result_values.push(val),
1462                Err(e) => {
1463                    vm.unpause_gc();
1464                    return Err(e);
1465                }
1466            }
1467        }
1468        vm.alloc_vector(result_values)
1469    };
1470
1471    Ok(result)
1472}
