13    new_total_width: usize,

15    new_num_lines: usize,

18fn calc_width_to_cursor<I>(
19    prompt_width: usize,
20    buf_widths: I,
21    terminal_width: usize,
22    last_arg_newline: bool,
23) -> usize
24where
25    I: Iterator<Item = usize>,
26{
27    let mut total = 0;
28    for line in buf_widths {
29        if total % terminal_width != 0 {
30            total = ((total / terminal_width) + 1) * terminal_width;
31        }
32        total += prompt_width + line;
33    }
34    if total == 0 {
35        // no lines returned, treat buf_width as 0
36        total += prompt_width
37    } else if last_arg_newline {
38        if total % terminal_width != 0 {
39            total = ((total / terminal_width) + 1) * terminal_width;
40        }
41        total += prompt_width;
42    }
43    total
44}

46impl Metrics {
47    pub fn new(
48        prompt: &str,
49        buf: &Buffer,
50        cursor: &Cursor,
51        autosuggestion: Option<&Buffer>,
52    ) -> io::Result<Self> {
53        let width = util::terminal_width()?;
54        let prompt_width = util::last_prompt_line_width(prompt);
55        let orig_buf_width = buf.line_widths();
56
57        let (buf_widths, buf_widths_end_newline) = match autosuggestion {
58            Some(suggestion) => (suggestion.line_widths(), suggestion.is_last_arg_newline()),
59            None => (orig_buf_width, buf.is_last_arg_newline()),
60        };
61
62        // Width of the current buffer lines (including autosuggestion) from the start to the cursor
63        let (buf_widths_to_cursor, buf_widths_cursor_end_newline) = match autosuggestion {
64            // Cursor might overrun autosuggestion with history search.
65            Some(suggestion) if cursor.curr_grapheme() < suggestion.num_graphemes() => (
66                suggestion.line_width_until(cursor.curr_grapheme()),
67                suggestion.is_last_arg_newline(),
68            ),
69            _ => (
70                buf.line_width_until(cursor.curr_grapheme()),
71                buf.is_last_arg_newline(),
72            ),
73        };
74
75        // Total number of terminal spaces taken up by prompt and buffer
76        let new_total_width =
77            calc_width_to_cursor(prompt_width, buf_widths, width, buf_widths_end_newline);
78        let new_total_width_to_cursor = calc_width_to_cursor(
79            prompt_width,
80            buf_widths_to_cursor,
81            width,
82            buf_widths_cursor_end_newline,
83        );
84
85        let new_num_lines = (new_total_width + width) / width;
86
87        Ok(Metrics {
88            width,
89            prompt_width,
90            new_total_width,
91            new_total_width_to_cursor,
92            new_num_lines,
93        })
94    }
95
96    pub fn term_cursor_line(&self) -> usize {
97        (self.new_total_width_to_cursor + self.width) / self.width
98    }
99
100    pub fn cursor_line_diff(&self) -> isize {
101        self.new_num_lines as isize - self.term_cursor_line() as isize
102    }
103
104    pub fn at_end_of_line(&self) -> bool {
105        self.new_total_width % self.width == 0
106    }
107
108    pub fn cursor_col_diff(&self) -> isize {
109        self.new_total_width as isize
110            - self.new_total_width_to_cursor as isize
111            - self.cursor_line_diff() * self.width as isize
112    }
113}

115/// An interface to simplify writing to and clearing the terminal. This is where most
116/// of the display logic for interfacing with the console library exists.
117pub struct Terminal<'a> {
118    out: &'a mut dyn io::Write,
119    // The line of the cursor relative to the prompt. 1-indexed.
120    // So if the cursor is on the same line as the prompt, `term_cursor_line == 1`.
121    // If the cursor is on the line below the prompt, `term_cursor_line == 2`.
122    term_cursor_line: usize,
123    // Last string that was colorized and last colorized version.
124    color_lines: Option<(String, String)>,
125    // A closure that is evaluated just before we write to out.
126    // This allows us to do custom syntax highlighting and other fun stuff.
127    closure: Option<ColorClosure>,
128    // Use the closure if it is set.
129    use_closure: bool,
130    buf: &'a mut String,
131}
132
133fn fmt_io_err(err: std::fmt::Error) -> io::Error {
134    let msg = format!("{}", err);
135    io::Error::new(io::ErrorKind::Other, msg)
136}

154    pub fn make_prompt(&mut self, prompt: Prompt) -> io::Result<Prompt> {
155        self.out.write_all("⏎".as_bytes())?;
156        for _ in 0..(util::terminal_width().unwrap_or(80) - 1) {
157            self.out.write_all(b" ")?; // if the line is not empty, overflow on next line
158        }
159        self.out.write_all("\r \r".as_bytes())?; // Erase the "⏎" if nothing overwrites it
160        let Prompt {
161            prefix,
162            mut prompt,
163            suffix,
164        } = prompt;
165        for (i, pline) in prompt.split('\n').enumerate() {
166            if i > 0 {
167                self.out.write_all(b"\r\n")?;
168            }
169            self.out.write_all(pline.as_bytes())?;
170        }
171        if let Some(index) = prompt.rfind('\n') {
172            prompt = prompt.split_at(index + 1).1.into()
173        }
174        Ok(Prompt {
175            prefix,
176            prompt,
177            suffix,
178        })
179    }

190    fn colorize(&mut self, line: &str) -> String {
191        match self.closure {
192            Some(ref mut f) if self.use_closure => {
193                let color = f(line);
194                self.color_lines = Some((line.to_string(), color.clone()));
195                color
196            }
197            Some(_) => {
198                if let Some((old_line, colorized)) = &self.color_lines {
199                    if line.starts_with(old_line) {
200                        let mut new_line = colorized.clone();
201                        new_line.push_str(&line[old_line.len()..]);
202                        new_line
203                    } else {
204                        line.to_owned()
205                    }
206                } else {
207                    line.to_owned()
208                }
209            }
210            _ => line.to_owned(),
211        }
212    }

214    fn display_with_suggest(
215        &mut self,
216        line: &str,
217        is_search: bool,
218        buf_num_remaining_bytes: usize,
219    ) -> io::Result<()> {
220        let start = self.colorize(&line[..buf_num_remaining_bytes]);
221        if is_search {
222            write!(self.buf, "{}", color::Yellow.fg_str()).map_err(fmt_io_err)?;
223        }
224        write!(self.buf, "{}", start).map_err(fmt_io_err)?;
225        if !is_search {
226            write!(self.buf, "{}", color::Yellow.fg_str()).map_err(fmt_io_err)?;
227        }
228        self.buf.push_str(&line[buf_num_remaining_bytes..]);
229        Ok(())
230    }

242    fn print_completion_list(
243        completions: &[String],
244        highlighted: Option<usize>,
245        output_buf: &mut String,
246    ) -> io::Result<usize> {
247        use std::cmp::max;
248
249        let (w, _) = sl_console::terminal_size()?;
250
251        // XXX wide character support
252        let max_word_size = completions.iter().fold(1, |m, x| max(m, x.chars().count()));
253        let cols = max(1, w as usize / (max_word_size));
254        let col_width = 2 + w as usize / cols;
255        let cols = max(1, w as usize / col_width);
256
257        let lines = completions.len() / cols;
258
259        let mut i = 0;
260        for (index, com) in completions.iter().enumerate() {
261            match i.cmp(&cols) {
262                Ordering::Greater => unreachable!(),
263                Ordering::Less => {}
264                Ordering::Equal => {
265                    output_buf.push_str("\r\n");
266                    i = 0;
267                }
268            }
269
270            if Some(index) == highlighted {
271                write!(
272                    output_buf,
273                    "{}{}",
274                    color::Black.fg_str(),
275                    color::White.bg_str()
276                )
277                .map_err(fmt_io_err)?;
278            }
279            write!(output_buf, "{:<1$}", com, col_width).map_err(fmt_io_err)?;
280            if Some(index) == highlighted {
281                write!(
282                    output_buf,
283                    "{}{}",
284                    color::Reset.bg_str(),
285                    color::Reset.fg_str()
286                )
287                .map_err(fmt_io_err)?;
288            }
289
290            i += 1;
291        }
292
293        Ok(lines)
294    }

296    pub fn clear_after_cursor(&mut self) -> io::Result<()> {
297        self.buf.push_str("\x1B[?1000l\x1B[?1l");
298
299        if self.term_cursor_line > 1 {
300            write!(self.buf, "{}", cursor::Up(self.term_cursor_line as u16 - 1))
301                .map_err(fmt_io_err)?;
302        }
303
304        write!(self.buf, "\r{}", clear::AfterCursor).map_err(fmt_io_err)
305    }

307    pub fn maybe_write_completions(
308        &mut self,
309        show_completions_hint: Option<&(Vec<String>, Option<usize>)>,
310    ) -> io::Result<usize> {
311        // If we're cycling through completions, show those
312        let mut completion_lines = 0;
313        if let Some((completions, i)) = show_completions_hint {
314            completion_lines = 1 + Self::print_completion_list(completions, *i, self.buf)?;
315            self.buf.push_str("\r\n");
316        }
317        Ok(completion_lines)
318    }

320    pub(crate) fn show_lines(
321        &mut self,
322        buf: &Buffer,
323        autosuggestion: Option<&Buffer>,
324        show_autosuggest: bool,
325        metrics: Metrics,
326        is_search: bool,
327    ) -> io::Result<()> {
328        // If we have an autosuggestion, we make the autosuggestion the buffer we print out.
329        // We get the number of bytes in the buffer (but NOT the autosuggestion).
330        // Then, we loop and subtract from that number until it's 0, in which case we are printing
331        // the autosuggestion from here on (in a different color).
332        let (lines, lines_len) = match autosuggestion {
333            Some(suggestion) if show_autosuggest => (suggestion.lines(), suggestion.num_lines()),
334            _ => (buf.lines(), buf.num_lines()),
335        };
336        let mut buf_num_remaining_bytes = buf.num_bytes();
337
338        for (i, line) in lines.enumerate() {
339            if i > 0 {
340                write!(self.buf, "{}", cursor::Right(metrics.prompt_width as u16))
341                    .map_err(fmt_io_err)?;
342            }
343
344            if buf_num_remaining_bytes == 0 {
345                self.buf.push_str(line);
346            } else if line.as_bytes().len() > buf_num_remaining_bytes {
347                self.display_with_suggest(line, is_search, buf_num_remaining_bytes)?;
348                buf_num_remaining_bytes = 0;
349            } else {
350                buf_num_remaining_bytes -= line.as_bytes().len();
351                let written_line = self.colorize(line);
352                if is_search {
353                    write!(self.buf, "{}", color::Yellow.fg_str()).map_err(fmt_io_err)?;
354                }
355                self.buf.push_str(&written_line);
356            }
357            if i + 1 < lines_len {
358                self.buf.push_str("\r\n");
359            }
360        }
361        Ok(())
362    }

364    pub(crate) fn display(&mut self, metrics: Metrics, completion_lines: usize) -> io::Result<()> {
365        // at the end of the line, move the cursor down a line
366        if metrics.at_end_of_line() {
367            self.buf.push_str("\r\n");
368        }
369
370        // The term cursor is now on the bottom line. We may need to move the term cursor up
371        // to the line where the true cursor is.
372        let cursor_line_diff = metrics.cursor_line_diff();
373        match cursor_line_diff.cmp(&0) {
374            Ordering::Greater => {
375                write!(self.buf, "{}", cursor::Up(cursor_line_diff as u16)).map_err(fmt_io_err)?;
376            }
377            Ordering::Less => unreachable!(),
378            Ordering::Equal => {}
379        }
380
381        // Now that we are on the right line, we must move the term cursor left or right
382        // to match the true cursor.
383        let cursor_col_diff = metrics.cursor_col_diff();
384        match cursor_col_diff.cmp(&0) {
385            Ordering::Greater => {
386                write!(self.buf, "{}", cursor::Left(cursor_col_diff as u16)).map_err(fmt_io_err)?
387            }
388            Ordering::Less => {
389                write!(self.buf, "{}", cursor::Right((-cursor_col_diff) as u16))
390                    .map_err(fmt_io_err)?;
391            }
392            Ordering::Equal => {}
393        }
394
395        self.term_cursor_line = completion_lines + metrics.term_cursor_line();
396
397        write!(
398            self.buf,
399            "{}{}",
400            color::Reset.fg_str(),
401            color::Reset.bg_str()
402        )
403        .map_err(fmt_io_err)?;
404
405        self.out.write_all(self.buf.as_bytes())?;
406        self.buf.clear();
407        self.out.flush()?;
408
409        Ok(())
410    }

412    pub fn flush(&mut self) -> io::Result<()> {
413        self.out.flush()
