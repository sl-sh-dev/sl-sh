14/// Usage: (get-temp ["/path/to/directory/to/use/as/base" "optional-prefix" "optional-suffix" length])
15///
16/// Creates a directory inside of an OS specific temporary directory. See [temp-dir](root::temp-dir)
17/// for OS specific notes. Also accepts an optional prefix, an optional suffix, and an optional
18/// length for the random number of characters in the temporary file created. Defaults to prefix of
19/// ".tmp", no suffix, and five random characters.
20///
21/// Section: file
22///
23/// Example:
24/// (test::assert-true (str-contains (get-temp) (temp-dir)))
25///
26/// (with-temp (fn (tmp)
27///         (let (tmp-dir (get-temp tmp))
28///             (test::assert-true (str-contains tmp-dir tmp)))))
29///
30/// (with-temp (fn (tmp)
31///         (let (tmp-dir (get-temp tmp "some-prefix"))
32///             (test::assert-true (str-contains tmp-dir tmp))
33///             (test::assert-true (str-contains tmp-dir "some-prefix")))))
34///
35/// (with-temp (fn (tmp)
36///         (let (tmp-dir (get-temp tmp "some-prefix" "some-suffix"))
37///             (test::assert-true (str-contains tmp-dir tmp))
38///             (test::assert-true (str-contains tmp-dir "some-prefix"))
39///             (test::assert-true (str-contains tmp-dir "some-suffix")))))
40///
41/// (with-temp (fn (tmp)
42///         (let (tmp-dir (get-temp tmp "some-prefix" "some-suffix" 6))
43///             (test::assert-true (str-contains tmp-dir tmp))
44///             (test::assert-true (str-contains tmp-dir "some-prefix"))
45///             (test::assert-true (str-contains tmp-dir "some-suffix"))
46///             (test::assert-equal (len "some-prefix012345some-suffix") (len (fs-base tmp-dir))))))
47#[sl_sh_fn(fn_name = "get-temp")]
48fn get_temp(
49    path: Option<String>,
50    prefix: Option<String>,
51    suffix: Option<String>,
52    len: Option<i64>,
53) -> VMResult<String> {
54    let fn_name = "get-temp";
55    let dir = get_provided_or_default_temp(path, fn_name)?;
56    let prefix = prefix.as_deref().unwrap_or_default();
57    let suffix = suffix.as_deref().unwrap_or_default();
58    let len = len.unwrap_or(LEN);
59    let dir = create_temp_dir(dir.as_path(), prefix, suffix, len, fn_name)?;
60    if let Some(path) = dir.to_str() {
61        let path = path.to_string();
62        Ok(path)
63    } else {
64        let msg = format!("{} unable to provide temporary directory", fn_name);
65        Err(VMError::new("io", msg))
66    }
67}

69/// Usage: (temp-dir)
70///
71/// Returns a string representing the temporary directory. See [get-temp](root::get-temp) for higher
72/// level temporary directory creation mechanism.
73///
74/// On Unix:
75/// Returns the value of the TMPDIR environment variable if it is set, otherwise for non-Android it
76/// returns /tmp. If Android, since there is no global temporary folder (it is usually allocated
77/// per-app), it returns /data/local/tmp.
78///
79/// On Windows:
80/// Returns the value of, in order, the TMP, TEMP, USERPROFILE environment variable if any are set and
81/// not the empty string. Otherwise, temp_dir returns the path of the Windows directory. This behavior
82/// is identical to that of GetTempPath, which this function uses internally.
83///
84/// Section: file
85///
86/// Example:
87/// (test::assert-true (fs-dir? (temp-dir)))
88#[sl_sh_fn(fn_name = "temp-dir")]
89fn builtin_temp_dir() -> VMResult<String> {
90    if let Some(path) = temp_dir().to_str() {
91        Ok(path.to_string())
92    } else {
93        Err(VMError::new(
94            "io",
95            "temp-dir: unable to provide temporary directory".to_string(),
96        ))
97    }
98}

100fn get_provided_or_default_temp(path: Option<String>, fn_name: &str) -> VMResult<PathBuf> {
101    match path {
102        None => Ok(temp_dir()),
103        Some(path) => {
104            let dir = expand_tilde(path.into());
105            let p = dir.as_path();
106            if p.exists() && p.is_dir() {
107                Ok(dir)
108            } else {
109                let msg = format!(
110                    "{} unable to provide temporary file in provided directory",
111                    fn_name
112                );
113                Err(VMError::new("io", msg))
114            }
115        }
116    }
117}

119fn create_temp_dir(
120    path: &Path,
121    prefix: &str,
122    suffix: &str,
123    len: i64,
124    fn_name: &str,
125) -> VMResult<PathBuf> {
126    if path.exists() && path.is_dir() {
127        let dir_name = random_name(prefix, suffix, len);
128        let dir = Path::new::<OsStr>(dir_name.as_ref());
129        let dir = path.join(dir);
130        fs::create_dir(dir.as_path()).map_err(|err| {
131            let msg = format!("{} unable to create temporary directory inside default temporary directory ({:?}), reason: {:?}", fn_name, dir.as_path(), err);
132            VMError::new("io", msg)
133        })?;
134        Ok(dir)
135    } else {
136        let msg = format!("{} unable to provide temporary directory", fn_name);
137        Err(VMError::new("io", msg))
138    }
139}

141fn random_name(prefix: &str, suffix: &str, len: i64) -> String {
142    let prefix = if prefix.is_empty() { ".tmp" } else { prefix };
143    let mut rng = rand::thread_rng();
144    let name = rand_alphanumeric_str(len.unsigned_abs(), &mut rng);
145    format!("{}{}{}", prefix, name, suffix)
146}

152fn create_temp_file(
153    dir: PathBuf,
154    prefix: &Option<String>,
155    suffix: &Option<String>,
156    len: Option<i64>,
157    fn_name: &str,
158) -> VMResult<PathBuf> {
159    let prefix = prefix.as_ref().map(|x| x.as_str()).unwrap_or_default();
160    let suffix = suffix.as_ref().map(|x| x.as_str()).unwrap_or_default();
161    let len = len.unwrap_or(LEN);
162    let filename = random_name(prefix, suffix, len);
163    let p = Path::new::<OsStr>(filename.as_ref());
164    let file = dir.join(p);
165    let p = file.as_path();
166    File::create(p).map_err(|err| {
167        let msg = format!(
168            "{} unable to create temporary file inside temporary directory ({:?}), reason: {:?}",
169            fn_name,
170            dir.as_path(),
171            err
172        );
173        VMError::new("io", msg)
174    })?;
175    Ok(file)
176}

178/// Usage: (get-temp-file ["/path/to/directory/to/use/as/base" "optional-prefix" "optional-suffix" length])
179///
180/// Returns name of file created inside temporary directory. Optionally takes a directory to use as
181/// the parent directory of the temporary file. Also accepts an optional prefix, an optional suffix,
182/// and an optional length for the random number of characters in the temporary files created. Defaults
183/// to prefix of ".tmp", no suffix, and five random characters.
184///
185/// Section: file
186///
187/// Example:
188/// (test::assert-true (str-contains (get-temp-file) (temp-dir)))
189///
190/// (with-temp (fn (tmp)
191///         (let (tmp-file (get-temp-file tmp))
192///             (test::assert-true (str-contains tmp-file tmp)))))
193///
194/// (with-temp (fn (tmp)
195///         (let (tmp-file (get-temp-file tmp "some-prefix"))
196///             (test::assert-true (str-contains tmp-file "some-prefix")))))
197///
198/// (with-temp (fn (tmp)
199///         (let (tmp-file (get-temp-file tmp "some-prefix" "some-suffix"))
200///             (test::assert-true (str-contains tmp-file "some-prefix"))
201///             (test::assert-true (str-contains tmp-file "some-suffix")))))
202///
203/// (with-temp (fn (tmp)
204///         (let (tmp-file (get-temp-file tmp "some-prefix" "some-suffix" 10))
205///             (test::assert-true (str-contains tmp-file "some-prefix"))
206///             (test::assert-true (str-contains tmp-file "some-suffix"))
207///             (test::assert-equal (len "some-prefix0123456789some-suffix") (len (fs-base tmp-file))))))
208#[sl_sh_fn(fn_name = "get-temp-file")]
209fn get_temp_file(
210    path: Option<String>,
211    prefix: Option<String>,
212    suffix: Option<String>,
213    len: Option<i64>,
214) -> VMResult<String> {
215    let fn_name = "get-temp-file";
216    let dir = get_provided_or_default_temp(path, fn_name)?;
217    let file = create_temp_file(dir, &prefix, &suffix, len, fn_name)?;
218    if let Some(path) = file.as_path().to_str() {
219        Ok(path.to_string())
220    } else {
221        let msg = format!("{} unable to provide temporary file", fn_name);
222        Err(VMError::new("io", msg))
223    }
224}
