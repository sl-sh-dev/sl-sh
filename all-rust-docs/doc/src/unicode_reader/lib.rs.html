9//! This crate provides adaptors which wrap byte-oriented readers and yield the UTF-8 data as
10//! Unicode code points or grapheme clusters.
11//!
12//! Unlike other Unicode parsers which work on strings (e.g.
13//! [unicode_segmentation](https://crates.io/crates/unicode_segmentation) upon which this is built),
14//! this crate works on streams and doesn't require reading the entire data into memory. Instead it
15//! yields the graphemes or code points as it reads them.
16//!
17//! # Example
18//!
19//! ```rust
20//! extern crate unicode_reader;
21//! use unicode_reader::{CodePoints, Graphemes};
22//!
23//! use std::io::Cursor;
24//!
25//! fn main() {
26//!     let input = Cursor::new("He\u{302}\u{320}llo");
27//!     let mut graphemes = Graphemes::from(input);
28//!     assert_eq!("H",                 graphemes.next().unwrap().unwrap());
29//!     assert_eq!("e\u{302}\u{320}",   graphemes.next().unwrap().unwrap()); // note 3 characters
30//!     assert_eq!("l",                 graphemes.next().unwrap().unwrap());
31//!     assert_eq!("l",                 graphemes.next().unwrap().unwrap());
32//!     assert_eq!("o",                 graphemes.next().unwrap().unwrap());
33//!     assert!(graphemes.next().is_none());
34//!
35//!     let greek_bytes = vec![0xCE, 0xA7, 0xCE, 0xB1, 0xCE, 0xAF, 0xCF, 0x81, 0xCE, 0xB5,
36//!                            0xCF, 0x84, 0xCE, 0xB5];
37//!     let mut codepoints = CodePoints::from(Cursor::new(greek_bytes));
38//!     assert_eq!(vec!['Χ', 'α', 'ί', 'ρ', 'ε', 'τ', 'ε'],
39//!                 codepoints.map(|r| r.unwrap())
40//!                           .collect::<Vec<char>>());
41//! }
42//! ```
43//!
44//! [Repository](https://github.com/wfraser/rust_unicode_reader)
45//!
46//! [Documentation](https://docs.rs/unicode_reader)
47

57use std::io::{Bytes, Read};

59/// CodePoints can be constructed for any byte-oriented reader.
60impl<R: Read> From<R> for CodePoints<Bytes<R>> {
61    fn from(input: R) -> CodePoints<Bytes<R>> {
62        CodePoints::from(input.bytes())
63    }
64}
65
66/// Graphemes can be constructed for any byte-oriented reader by going through CodePoints as an
67/// internal layer.
68impl<R: Read> From<R> for Graphemes<CodePoints<Bytes<R>>> {
69    fn from(input: R) -> Graphemes<CodePoints<Bytes<R>>> {
70        Graphemes::from(CodePoints::from(input))
71    }
72}

93    fn assert_badutf8err<T>(reader: &mut dyn Iterator<Item = io::Result<T>>,
94                            kind: io::ErrorKind,
95                            bad_bytes: Vec<u8>) {
96        let result = reader.next().unwrap();
97        assert!(result.is_err());
98        let ioerr: io::Error = result.err().unwrap();
99        assert_eq!(kind, ioerr.kind());
100        let inner: Box<BadUtf8Error> = ioerr.into_inner().unwrap().downcast().unwrap();
101        assert_eq!(bad_bytes, inner.bytes);
102    }

104    #[test]
105    fn test_utf8_error() {
106        let bad = b"a\xe2\x28\xa1bc"; // the 2nd byte of the 3-byte set is invalid utf8
107        let mut codepoints = CodePoints::from(Cursor::new(bad));
108        assert_eq!('a', codepoints.next().unwrap().unwrap());
109
110        // Here it should fail to parse a code point.
111        assert_badutf8err(&mut codepoints,
112                          io::ErrorKind::InvalidData,
113                          vec![0xe2, 0x28, 0xa1]);
114
115        // It should recover and continue after the bad bytes.
116        assert_eq!('b', codepoints.next().unwrap().unwrap());
117        assert_eq!('c', codepoints.next().unwrap().unwrap());
118        assert!(codepoints.next().is_none());
119    }

121    #[test]
122    fn test_error_at_end() {
123        let bad = b"a\xe2\x80"; // ends in incomplete 3-byte UTF-8
124        let mut codepoints = CodePoints::from(Cursor::new(bad));
125        assert_eq!('a', codepoints.next().unwrap().unwrap());
126
127        // Here it should fail to parse a code point.
128        assert_badutf8err(&mut codepoints,
129                          io::ErrorKind::UnexpectedEof,
130                          vec![0xe2, 0x80]);
131
132        // After reading the bad bytes, it should report the end of stream.
133        assert!(codepoints.next().is_none());
134    }

136    #[test]
137    fn test_grapheme_bad_utf8() {
138        let bad = b"ab\xe2\x28\xa1cd";
139        let mut graphemes = Graphemes::from(Cursor::new(bad));
140
141        assert_eq!("a", graphemes.next().unwrap().unwrap());
142
143        // Graphemes will hit the error here, but it needs to store it and return its pending
144        // buffer instead.
145        assert_eq!("b", graphemes.next().unwrap().unwrap());
146
147        // Now it should raise the error.
148        assert_badutf8err(&mut graphemes,
149                          io::ErrorKind::InvalidData,
150                          vec![0xe2, 0x28, 0xa1]);
151
152        // Now it should recover and return more valid data.
153        assert_eq!("c", graphemes.next().unwrap().unwrap());
154        assert_eq!("d", graphemes.next().unwrap().unwrap());
155        assert!(graphemes.next().is_none());
156    }

174    #[test]
175    fn test_notoria() {
176        let bad = [0x80, 1, 2, 3, 4, 5, 6, 7];
177        let mut codepoints = CodePoints::from(&bad[..]);
178
179        assert_badutf8err(&mut codepoints,
180            io::ErrorKind::InvalidData,
181            vec![0x80]);
182
183        for i in 1 ..= 7 {
184            match codepoints.next() {
185                Some(Ok(c)) if c == char::from(i) => (),
186                other => panic!("at {}, expected '{:?}', found: {:?}", i, char::from(i), other),
187            }
188        }
189        assert!(codepoints.next().is_none());
190    }
