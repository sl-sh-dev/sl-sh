1use bridge_adapters::add_builtin;
2use compile_state::state::SloshVm;
3use sl_compiler::Reader;
4use sl_compiler::load_eval::{CORE_LISP_NAME, TEST_LISP_NAME, run_reader};
5use slosh_lib::{load_builtins_lisp_less_sloshrc, run};
6use slosh_test_lib::docs;
7use slvm::{VMError, VMResult, Value};
8use std::process;
9
10pub const VERSION_STRING: &str = env!("VERSION_STRING");
11
12fn version(vm: &mut SloshVm, registers: &[Value]) -> VMResult<Value> {
13    if !registers.is_empty() {
14        return Err(VMError::new_compile("version: requires no argument"));
15    }
16    Ok(vm.alloc_string(VERSION_STRING.to_string()))
17}
18
19fn modify_vm(vm: &mut SloshVm) {
20    docs::add_builtins(vm);
21
22    add_builtin(
23        vm,
24        "version",
25        version,
26        "Return the software version string.",
27    );
28
29    let code = format!(
30        "(do {})",
31        [CORE_LISP_NAME, TEST_LISP_NAME]
32            .iter()
33            .map(|x| format!("(load \"{x}\")"))
34            .collect::<Vec<String>>()
35            .join(" ")
36    );
37    let mut reader = Reader::from_string(code, vm, "", 1, 0);
38    _ = run_reader(&mut reader);
39    _ = load_builtins_lisp_less_sloshrc(vm);
40}
41
42fn main() {
43    let exit_code = run(modify_vm);
44    process::exit(exit_code)
45}

47#[cfg(test)]
48mod tests {
49    use compiler_test_utils::exec;
50    use slosh_lib::{ENV, set_builtins_and_shell_builtins, set_initial_load_path};
51    use slvm::{Value, from_i56};
52    use std::fs::{File, create_dir_all};
53    use std::io::Write;
54    use std::ops::DerefMut;
55    use temp_env;
56    use tempfile::TempDir;
57
58    #[test]
59    fn test_load_path_no_home() {
60        // create home dir
61        let tmp_dir = TempDir::with_prefix("test_load_path").unwrap();
62        let home_dir = tmp_dir.path().to_str();
63        let home_path = home_dir.unwrap().to_string();
64
65        let tmp_0 = tmp_dir.path().join("tmp_0");
66        let tmp_1 = tmp_dir.path().join("tmp_1");
67        {
68            // create a dir with an add fcn that adds 1 in  add.slosh
69            create_dir_all(tmp_0.clone()).unwrap();
70            let file_0 = tmp_0.as_path().join("add.slosh");
71            let mut file_0 = File::create(file_0).unwrap();
72            writeln!(file_0, "(def add (fn (x) (+ 1 x)))").unwrap();
73            File::flush(&mut file_0).unwrap();
74
75            // create a dir with an add fcn that adds 2 in add.slosh
76            create_dir_all(tmp_1.clone()).unwrap();
77            let file_1 = tmp_1.as_path().join("add.slosh");
78            let mut file_1 = File::create(file_1).unwrap();
79            writeln!(file_1, "(def add (fn (x) (+ 2 x)))").unwrap();
80            File::flush(&mut file_1).unwrap();
81        }
82
83        let v = temp_env::with_var("HOME", home_dir, || {
84            ENV.with(|env| {
85                let mut vm = env.borrow_mut();
86                set_builtins_and_shell_builtins(vm.deref_mut());
87                set_initial_load_path(
88                    vm.deref_mut(),
89                    vec![
90                        &home_path,
91                        tmp_0.to_str().unwrap().as_ref(),
92                        tmp_1.to_str().unwrap().as_ref(),
93                    ],
94                );
95                _ = exec(vm.deref_mut(), "(load \"add.slosh\")");
96                let v = exec(vm.deref_mut(), "(add 1)");
97                match v {
98                    Value::Int(i) => from_i56(&i),
99                    _ => {
100                        panic!("Value should be an integer");
101                    }
102                }
103            })
104        });
105        assert_eq!(v, 2i64);
106    }
107}
