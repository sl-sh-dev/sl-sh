5fn compile_params(
6    env: &mut SloshVm,
7    state: &mut CompileState,
8    cdr: &[Value],
9    result: usize,
10    tail: bool,
11) -> VMResult<()> {
12    if !cdr.is_empty() {
13        for (i, r) in cdr.iter().enumerate() {
14            compile(env, state, *r, result + i)?;
15        }
16        let line = env.own_line();
17        if tail {
18            state
19                .chunk
20                .encode3(BMOV, 1, result as u16, cdr.len() as u16, line)?;
21        }
22    }
23    Ok(())
24}

26pub(crate) fn compile_call(
27    env: &mut SloshVm,
28    state: &mut CompileState,
29    callable: Value,
30    cdr: &[Value],
31    result: usize,
32) -> VMResult<()> {
33    let b_reg = result + cdr.len() + 1;
34    if b_reg > state.max_regs {
35        state.max_regs = b_reg;
36    }
37    let const_i = state.add_constant(callable);
38    let tail = state.tail && state.defers == 0;
39    state.tail = false;
40    compile_params(env, state, cdr, result + 1, tail)?;
41    let line = env.own_line();
42    state
43        .chunk
44        .encode2(CONST, b_reg as u16, const_i as u16, line)?;
45    if tail {
46        state
47            .chunk
48            .encode2(TCALL, b_reg as u16, cdr.len() as u16, line)?;
49    } else {
50        state
51            .chunk
52            .encode3(CALL, b_reg as u16, cdr.len() as u16, result as u16, line)?;
53    }
54    Ok(())
55}

57pub(crate) fn compile_callg(

74    }

78pub(crate) fn compile_call_reg(
79    env: &mut SloshVm,
80    state: &mut CompileState,
81    reg: u16,
82    cdr: &[Value],
83    result: usize,
84) -> VMResult<()> {
85    let tail = state.tail && state.defers == 0;
86    state.tail = false;
87    if tail {
88        if result as u16 + 1 != reg {
89            state
90                .chunk
91                .encode2(MOV, result as u16 + 1, reg, env.own_line())?;
92        }
93        // Lie about this being a tail call because we can not emit the BMOV yet.
94        compile_params(env, state, cdr, result + 2, false)?;
95        let b_reg = result + cdr.len() + 3;
96        if b_reg > state.max_regs {
97            state.max_regs = b_reg;
98        }
99        if result + 1 != b_reg {
100            state
101                .chunk
102                .encode2(MOV, b_reg as u16, result as u16 + 1, env.own_line())?;
103        }
104        state
105            .chunk
106            .encode3(BMOV, 1, result as u16 + 2, cdr.len() as u16, env.own_line())?;
107        state
108            .chunk
109            .encode2(TCALL, b_reg as u16, cdr.len() as u16, env.own_line())?;
110    } else {
111        compile_params(env, state, cdr, result + 1, tail)?;
112        state
113            .chunk
114            .encode3(CALL, reg, cdr.len() as u16, result as u16, env.own_line())?;
115    }
116    Ok(())
117}

119pub(crate) fn compile_call_myself(
120    env: &mut SloshVm,
121    state: &mut CompileState,
122    cdr: &[Value],
123    result: usize,
124    force_tail: bool,
125) -> VMResult<()> {
126    let tail = force_tail || (state.tail && state.defers == 0);
127    state.tail = false;
128    compile_params(env, state, cdr, result + 1, tail)?;
129    let line = env.own_line();
130    if tail {
131        state.chunk.encode1(TCALLM, cdr.len() as u16, line)?;
132    } else {
133        state
134            .chunk
135            .encode2(CALLM, cdr.len() as u16, result as u16, line)?;
136    }
137    Ok(())
