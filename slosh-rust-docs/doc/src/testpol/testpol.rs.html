1use std::sync::Arc;
2
3use slvm::opcodes::*;
4use slvm::{Chunk, VMError, Vm};
5
6fn main() -> Result<(), VMError> {
7    // algorithm from http://dan.corlan.net/bench.html
8    // Do a lot of loops and simple math and to see how we stack up.
9    /*
10    (defn eval-pol (n x)
11      (let ((su 0.0) (mu 10.0) (pu 0.0)
12            (pol (make-vec 100 0.0)))
13        (dotimes-i i n
14          (do
15            (set! su 0.0)
16            (dotimes-i j 100
17               (do
18                 (set! mu (/ (+ mu 2.0) 2.0))
19                 (vec-set! pol j mu)))
20            (dotimes-i j 100
21              (set! su (+ (vec-nth pol j) (* su x))))
22            (set! pu (+ pu su))))
23        (println pu)))
24             */
25    let mut vm = Vm::new();
26    //vm.pause_gc();
27    let mut chunk = Chunk::new("no_file", 1);
28    chunk.extra_regs = 150;
29    let n = chunk.add_constant(500_000.into()) as u16;
30    let x = chunk.add_constant(0.2.into()) as u16;
31    let su = chunk.add_constant(0.0.into()) as u16;
32    let mu = chunk.add_constant(10.0.into()) as u16;
33    let pu = chunk.add_constant(0.0.into()) as u16;
34    let zero = chunk.add_constant(0.into()) as u16;
35    //let five_hundred = chunk.add_constant(Value::Int(500)) as u16;
36    let zerof = chunk.add_constant(0.0.into()) as u16;
37    let twof = chunk.add_constant(2.0.into()) as u16;
38    let hundred = chunk.add_constant(100.into()) as u16;
39    //let hundred = chunk.add_constant(Value::Int(600)) as u16;
40    let one = chunk.add_constant(1.into()) as u16;
41    chunk.encode2(CONST, 1, n, Some(1))?;
42    chunk.encode2(CONST, 2, x, None)?;
43    chunk.encode2(CONST, 3, su, None)?;
44    chunk.encode2(CONST, 4, mu, None)?;
45    chunk.encode2(CONST, 5, pu, None)?;
46    chunk.encode2(CONST, 6, zero, None)?; // i
47    chunk.encode2(CONST, 7, zero, None)?; // j
48    //chunk.encode2(CONST, 7, five_hundred, None)?; // j
49    chunk.encode2(CONST, 8, twof, None)?; // 2.0
50    chunk.encode2(CONST, 100, hundred, None)?;
51    chunk.encode2(CONST, 101, one, None)?;
52    chunk.encode2(CONST, 103, zerof, None)?;
53
54    chunk.encode3(VECMKD, 10, 100, 103, None)?; // pols
55    //chunk.encode2(VECELS, 10, 100, None)?;
56    // loop i .. n
57    chunk.add_jump(chunk.code.len() as u32);
58    chunk.encode2(CONST, 3, zerof, None)?;
59    chunk.encode2(CONST, 7, zero, None)?; // j
60    //chunk.encode2(CONST, 7, five_hundred, None)?; // j
61    // loop j .. 100
62    // (set! mu (/ (+ mu 2.0) 2.0))
63    chunk.add_jump(chunk.code.len() as u32);
64    chunk.encode2(ADD, 4, 8, None)?;
65    chunk.encode2(DIV, 4, 8, None)?;
66    // (vec-set! pol j mu)))
67    chunk.encode3(SETCOL, 4, 10, 7, None)?;
68    //chunk.encode2(MOVI, 7, 4, None)?;
69
70    chunk.encode2(INC, 7, 1, None)?;
71    chunk.encode3(JMPLT, 7, 100, 1, None)?;
72    //chunk.encode_jump_offset(-19)?;
73    //chunk.encode_jump_offset(-18)?;
74
75    chunk.encode2(CONST, 7, zero, None)?; // j
76    //chunk.encode2(CONST, 7, five_hundred, None)?; // j
77    // (dotimes-i j 100 (j2)
78    //   (set! su (+ (vec-nth pol j) (* su x))))
79    chunk.add_jump(chunk.code.len() as u32);
80    chunk.encode2(MUL, 3, 2, None)?;
81    chunk.encode3(GET, 51, 10, 7, None)?;
82    //chunk.encode2(MOVII, 51, 7, None)?;
83    chunk.encode2(ADD, 3, 51, None)?;
84
85    chunk.encode2(INC, 7, 1, None)?;
86    chunk.encode3(JMPLT, 7, 100, 2, None)?;
87    //chunk.encode_jump_offset(-19)?;
88    //chunk.encode_jump_offset(-18)?;
89    // (set! pu (+ pu su))))
90    chunk.encode2(ADD, 5, 3, None)?;
91
92    chunk.encode2(INC, 6, 1, None)?;
93    chunk.encode3(JMPLT, 6, 1, 0, None)?;
94    //chunk.encode_jump_offset(-59)?;
95    //chunk.encode_jump_offset(-57)?;
96
97    chunk.encode1(SRET, 5, None)?;
98
99    let chunk = Arc::new(chunk);
100    match vm.execute(chunk.clone()) {
101        Err(e) => {
102            println!("ERROR: {e}");
103            chunk
104                .disassemble_chunk(&vm, 0)
105                .expect("failed to disassemble chunk");
106            println!("STACK: {:?}", &vm.stack_slice()[..vm.stack_max()]);
107        }
108        Ok(result) => {
109            println!("{result:?}");
110        }
111    }
112
113    Ok(())
114}
