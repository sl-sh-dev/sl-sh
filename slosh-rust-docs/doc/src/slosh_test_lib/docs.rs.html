114enum Namespace {

116    Other(Interned),

119impl Namespace {
120    fn display(&self, vm: &mut SloshVm) -> String {
121        match self {
122            Namespace::Global => GLOBAL_NAMESPACE.to_string(),
123            Namespace::Other(s) => {
124                let s = vm.get_interned(*s);
125                s.to_string()
126            }
127        }
128    }
129
130    fn get_doc(
131        &self,
132        interned: &Interned,
133        docs: &mut Vec<SloshDoc>,
134        vm: &mut SloshVm,
135        require_proper_format: bool,
136    ) -> DocResult<()> {
137        let slosh_doc = SloshDoc::new(*interned, vm, self.clone());
138        match slosh_doc {
139            Ok(slosh_doc) => {
140                docs.push(slosh_doc);
141            }
142            Err(e) => match e {
143                _ if !require_proper_format => {
144                    let incomplete_doc = SloshDoc::new_incomplete(*interned, vm, self.clone())?;
145                    docs.push(incomplete_doc);
146                }
147                _ => {
148                    return Err(e);
149                }
150            },
151        }
152        Ok(())
153    }
154
155    fn add_docs(
156        &self,
157        docs: &mut Vec<SloshDoc>,
158        vm: &mut SloshVm,
159        require_proper_format: bool,
160    ) -> DocResult<()> {
161        match self {
162            Namespace::Global => {
163                for g in vm.globals().clone().keys() {
164                    self.get_doc(g, docs, vm, require_proper_format)?;
165                }
166            }
167            Namespace::Other(i) => {
168                let value = builtins::retrieve_in_namespace(vm, i);
169                for v in value {
170                    if let Value::Symbol(sym) = v {
171                        self.get_doc(&sym, docs, vm, require_proper_format)?;
172                    }
173                }
174            }
175        }
176        docs.sort();
177        docs.dedup();
178        Ok(())
179    }

213impl DocStringSection {
214    pub fn from_symbol(slot: u32, sym: Value, vm: &mut SloshVm) -> DocResult<DocStringSection> {
215        let sym_str = sym.display_value(vm);
216        let raw_doc_string = Self::raw_docstring(slot, vm);
217        let backup_usage = slosh_lib::usage(vm, slot, &sym);
218        Self::parse_doc_string(Cow::Owned(sym_str), raw_doc_string, backup_usage)
219    }

221    fn raw_docstring(slot: u32, vm: &mut SloshVm) -> String {
222        let docstring_key = vm.intern_static("doc-string");
223        vm.get_global_property(slot, docstring_key)
224            .and_then(|x| match x {
225                Value::String(h) => Some(vm.get_string(h).to_string()),
226                Value::StringConst(i) => Some(vm.get_interned(i).to_string()),
227                _ => None,
228            })
229            // return default empty string and have parse_doc_string handle error if no doc provided.
230            .unwrap_or_default()
231    }

233    /// Given the rules for parsing slosh docstrings, parse one! See [`static@DOC_REGEX`]
234    /// for the specification.
235    pub fn parse_doc_string(
236        symbol: Cow<'_, str>,
237        raw_doc_string: String,
238        backup_usage: String,
239    ) -> DocResult<DocStringSection> {
240        let cap =
241            DOC_REGEX
242                .captures(raw_doc_string.as_str())
243                .ok_or_else(|| DocError::NoDocString {
244                    symbol: symbol.to_string(),
245                });
246        if EXEMPTIONS.contains(symbol.as_ref()) && cap.is_err() {
247            let usage = Some("unknown".to_string());
248            let description = "unknown".to_string();
249            let section = "undocumented".to_string();
250            let example = None;
251            return Ok(DocStringSection {
252                usage,
253                description,
254                section,
255                example,
256            });
257        }
258        let cap = cap?;
259        let mut usage = cap.get(2).map(|x| x.as_str().trim().to_string());
260        if usage.is_none() && !backup_usage.trim().is_empty() {
261            usage = Some(backup_usage);
262        }
263        let description = cap
264            .get(3)
265            .ok_or_else(|| DocError::DocStringMissingSection {
266                symbol: symbol.to_string(),
267                section: "Description".to_string(),
268            })
269            .map(|x| x.as_str().to_string())?;
270        let section = cap
271            .get(4)
272            .ok_or_else(|| DocError::DocStringMissingSection {
273                symbol: symbol.to_string(),
274                section: "Section".to_string(),
275            })
276            .map(|x| x.as_str().trim().to_string())?;
277        let example = cap.get(6).map(|x| x.as_str().trim().to_string());
278        if EXEMPTIONS.contains(symbol.as_ref()) {
279            Err(DocError::RemoveExemption {
280                symbol: symbol.to_string(),
281            })
282        } else {
283            Ok(DocStringSection {
284                usage,
285                description,
286                section,
287                example,
288            })
289        }
290    }

476enum DocError {

478    NoDocString { symbol: String },
479    RemoveExemption { symbol: String },
480    DocStringMissingSection { symbol: String, section: String },
481}

489impl Display for DocError {
490    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
491        let str = match self {
492            DocError::NoSymbol { symbol } => {
493                format!(
494                    "No documentation string provided for symbol {symbol}."
495                )
496            }
497            DocError::NoDocString { symbol } => {
498                format!(
499                    "Either documentation provided does not conform to conventional layout or no documentation string provided for symbol {symbol} all slosh functions with documentation must have a string that conforms to the conventional layout."
500                )
501            }
502            DocError::DocStringMissingSection { symbol, section } => {
503                format!("Invalid documentation string for symbol {symbol}, missing required section {section:?}")
504            }
505            DocError::RemoveExemption { symbol} => {
506                format!("Documentation has been added for {symbol}, remove it from EXEMPTIONS list in slosh_test::docs::EXEMPTIONS.")
507            }
508        }
509            .to_string();
510        write!(f, "{}", str)
511    }

551        let use_background_key = Value::Keyword(vm.intern("use-background"));

567                Value::StringConst(i) => options.default_color = vm.get_interned(i).to_string(),
568                _ => {}

570        }

580fn highlight_matches(text: &str, query: &DocSearchQuery, style: &StyleOptions) -> String {
581    if query.use_regex {
582        highlight_regex_matches(text, &query.query, style)
583    } else if query.basic {
584        highlight_basic_matches(text, &query.query, style)
585    } else {
586        highlight_substring_matches(text, &query.query, style)
587    }
588}

590fn highlight_substring_matches(text: &str, query: &str, style: &StyleOptions) -> String {
591    let text_lower = text.to_lowercase();
592    let query_lower = query.to_lowercase();
593    let mut result = String::new();
594    let mut last_end = 0;
595
596    for (start, _) in text_lower.match_indices(&query_lower) {
597        // Add text before match
598        result.push_str(&text[last_end..start]);
599        // Add highlighted match
600        result.push_str(&style.highlight_color);
601        result.push_str(&text[start..start + query.len()]);
602        result.push_str(&style.default_color);
603        last_end = start + query.len();
604    }
605
606    // Add remaining text
607    result.push_str(&text[last_end..]);
608    result
609}

611fn highlight_regex_matches(text: &str, pattern: &str, style: &StyleOptions) -> String {
612    if let Ok(re) = Regex::new(pattern) {
613        let mut result = String::new();
614        let mut last_end = 0;
615
616        for mat in re.find_iter(text) {
617            // Add text before match
618            result.push_str(&text[last_end..mat.start()]);
619            // Add highlighted match
620            result.push_str(&style.highlight_color);
621            result.push_str(mat.as_str());
622            result.push_str(&style.default_color);
623            last_end = mat.end();
624        }
625
626        // Add remaining text
627        result.push_str(&text[last_end..]);
628        result
629    } else {
630        text.to_string()
631    }
632}

634fn highlight_basic_matches(text: &str, pattern: &str, style: &StyleOptions) -> String {
635    let pattern_lower = pattern.to_lowercase();
636    let mut result = String::new();
637    let mut pattern_chars = pattern_lower.chars().peekable();
638    let mut in_match = false;
639
640    for ch in text.chars() {
641        if let Some(&pattern_ch) = pattern_chars.peek() {
642            if ch.to_lowercase().next() == Some(pattern_ch) {
643                if !in_match {
644                    result.push_str(&style.highlight_color);
645                    in_match = true;
646                }
647                result.push(ch);
648                pattern_chars.next();
649            } else {
650                if in_match {
651                    result.push_str(&style.default_color);
652                    in_match = false;
653                }
654                result.push(ch);
655            }
656        } else {
657            if in_match {
658                result.push_str(&style.default_color);
659                in_match = false;
660            }
661            result.push(ch);
662        }
663    }
664
665    if in_match {
666        result.push_str(&style.default_color);
667    }
668
669    result
670}

672fn generate_styled_output(
673    _vm: &mut SloshVm,
674    results: &BTreeSet<SloshDoc>,
675    query: &DocSearchQuery,
676    style: &StyleOptions,
677) -> String {
678    let mut output = String::new();
679    let len = results.len();
680
681    for (i, doc) in results.iter().enumerate() {
682        if i > 0 {
683            output.push_str("\n\n\n");
684        }
685
686        // Result header with highlighted symbol
687        output.push_str(&format!("Result {}: ", i + 1));
688        let mut doc = doc.clone();
689        doc = doc.into_styled_output(query, style);
690        output.push_str(&format!("{doc}"));
691    }
692
693    output.push_str(&format!(
694        "\nFound {} result{}.",
695        len,
696        if len == 1 { "" } else { "s" }
697    ));
698
699    output

767/// Each doc has a tag in its `Section:` definition by convention that logically groups functions.
768/// Using a HashMap store the section tags as keys and add all slosh docs from to a vector as a value
769/// corresponding to its section.
770fn get_docs_by_section(
771    vm: &mut SloshVm,
772    require_proper_format: bool,
773) -> HashMap<String, BTreeSet<SloshDoc>> {
774    let mut docs_by_section: HashMap<String, BTreeSet<SloshDoc>> = HashMap::new();
775    let mut docs: Vec<SloshDoc> = vec![];
776    Namespace::Global
777        .add_docs(&mut docs, vm, require_proper_format)
778        .unwrap();
779    let namespaces = builtins::get_namespaces_interned(vm);
780    for i in namespaces {
781        let namespace = Namespace::Other(i);
782        namespace
783            .add_docs(&mut docs, vm, require_proper_format)
784            .unwrap();
785    }
786    for d in docs {
787        let d = d.clone();
788        let section = d.doc_string.section.clone();
789        docs_by_section.entry(section).or_default().insert(d);
790    }
791    docs_by_section
792}

794fn build_symbols_list(
795    docs_by_section: &BTreeMap<String, BTreeSet<SloshDoc>>,
796    namer: fn(&str, &SloshDoc) -> String,
797) -> BTreeMap<String, String> {
798    let mut map = BTreeMap::new();
799    for (section, v) in docs_by_section.iter() {
800        let mut list = "".to_string();
801        let len = v.len();
802        for (i, docs) in v.iter().enumerate() {
803            let name = namer(section.as_ref(), docs);
804            list += &name;
805
806            if i + 1 != len {
807                list += ", ";
808            }
809        }
810        list += "\n";
811        map.insert(section.to_string(), list);
812    }
813    map
814}

1152        self.use_regex = false;

1176        if self.use_regex {
1177            if let Ok(re) = Regex::new(&self.query) {
1178                re.is_match(text)
1179            } else {
1180                false
1181            }
1182        } else if self.basic {
1183            basic_match(text, &self.query)
1184        } else {
1185            text.to_lowercase().contains(&self.query.to_lowercase())
1186        }
1187    }

1190fn basic_match(text: &str, pattern: &str) -> bool {
1191    let text_lower = text.to_lowercase();
1192    let pattern_lower = pattern.to_lowercase();
1193    let mut pattern_chars = pattern_lower.chars();
1194    let mut current_char = pattern_chars.next();
1195
1196    for text_char in text_lower.chars() {
1197        if let Some(p_char) = current_char {
1198            if text_char == p_char {
1199                current_char = pattern_chars.next();
1200            }
1201        } else {
1202            return true;
1203        }
1204    }
1205
1206    current_char.is_none()
1207}

1209fn search_docs(
1210    vm: &mut SloshVm,
1211    query: &DocSearchQuery,
1212    require_proper_format: bool,
1213) -> BTreeSet<SloshDoc> {
1214    let docs_by_section = get_docs_by_section(vm, require_proper_format);
1215    let mut results = BTreeSet::new();
1216
1217    for (section, docs) in docs_by_section {
1218        // Check section filter
1219        if let Some(ref section_filter) = query.section_filter {
1220            if !query.matches_text(&section) && section != *section_filter {
1221                continue;
1222            }
1223        }
1224
1225        for doc in docs {
1226            // Check namespace filter
1227            if let Some(ref ns_filter) = query.namespace_filter {
1228                if !doc.namespace.contains(ns_filter) && doc.namespace != *ns_filter {
1229                    continue;
1230                }
1231            }
1232
1233            let mut matched = false;
1234
1235            // Search in symbol name
1236            if query.should_search_field("name")
1237                || query.should_search_field("symbol") && query.matches_text(&doc.symbol)
1238            {
1239                matched = true;
1240            }
1241
1242            // Search in usage
1243            if !matched
1244                && query.should_search_field(USAGE)
1245                && let Some(ref usage) = doc.doc_string.usage
1246                && query.matches_text(usage)
1247            {
1248                matched = true;
1249            }
1250
1251            // Search in description
1252            if !matched
1253                && query.should_search_field(DESCRIPTION)
1254                && query.matches_text(&doc.doc_string.description)
1255            {
1256                matched = true;
1257            }
1258
1259            // Search in section
1260            if !matched
1261                && query.should_search_field(SECTION)
1262                && query.matches_text(&doc.doc_string.section)
1263            {
1264                matched = true;
1265            }
1266
1267            // Search in example
1268            if !matched
1269                && query.should_search_field(EXAMPLE)
1270                && let Some(ref example) = doc.doc_string.example
1271                && query.matches_text(example)
1272            {
1273                matched = true;
1274            }
1275
1276            // If no specific fields specified, search all fields
1277            if !matched && query.fields.is_empty() {
1278                matched = query.matches_text(&doc.symbol)
1279                    || doc
1280                        .doc_string
1281                        .usage
1282                        .as_ref()
1283                        .map(|u| query.matches_text(u))
1284                        .unwrap_or(false)
1285                    || query.matches_text(&doc.doc_string.description)
1286                    || query.matches_text(&doc.doc_string.section)
1287                    || doc
1288                        .doc_string
1289                        .example
1290                        .as_ref()
1291                        .map(|e| query.matches_text(e))
1292                        .unwrap_or(false);
1293            }
1294
1295            if matched {
1296                results.insert(doc);
1297            }
1298        }
1299    }
1300
1301    results
1302}

1304fn doc_search(vm: &mut SloshVm, registers: &[Value]) -> VMResult<Value> {
1305    if registers.is_empty() {
1306        return Err(VMError::new_vm(
1307            "doc-search requires at least a query string".to_string(),
1308        ));
1309    }
1310
1311    let query_str = match &registers[0] {
1312        Value::String(s) => vm.get_string(*s).to_string(),
1313        Value::StringConst(i) => vm.get_interned(*i).to_string(),
1314        _ => {
1315            return Err(VMError::new_vm(
1316                "First argument must be a query string".to_string(),
1317            ));
1318        }
1319    };
1320
1321    let mut query = DocSearchQuery::new(query_str);
1322
1323    // Parse options if provided as a hash map
1324    let mut use_styled = true;
1325    let mut style_options = StyleOptions::default();
1326    let mut style_map_handle_opt = None;
1327    if registers.len() > 1 {
1328        match &registers[1] {
1329            Value::Map(handle) => {
1330                // Intern all keys first to avoid borrowing issues
1331                let regex_key = Value::Keyword(vm.intern_static("regex"));
1332                let basic_search_key = Value::Keyword(vm.intern_static("basic"));
1333                let fields_key = Value::Keyword(vm.intern_static("fields"));
1334                let ns_key = Value::Keyword(vm.intern_static("namespace"));
1335                let section_key = Value::Keyword(vm.intern_static("section"));
1336                let styled_key = Value::Keyword(vm.intern_static("styled"));
1337                let style_key = Value::Keyword(vm.intern_static("style"));
1338
1339                let map = vm.get_map(*handle);
1340
1341                // Check for regex option
1342                if let Some(Value::True) = map.get(vm, regex_key) {
1343                    query = query.with_regex();
1344                }
1345
1346                // Check for basic option
1347                if let Some(Value::True) = map.get(vm, basic_search_key) {
1348                    query = query.with_basic();
1349                }
1350
1351                // Check for fields option
1352                if let Some(Value::Vector(v_handle)) = map.get(vm, fields_key) {
1353                    let vec = vm.get_vector(v_handle);
1354                    let mut fields = Vec::new();
1355                    for val in vec.iter() {
1356                        match val {
1357                            Value::String(s) => fields.push(vm.get_string(*s).to_string()),
1358                            Value::StringConst(i) => fields.push(vm.get_interned(*i).to_string()),
1359                            Value::Keyword(i) => fields.push(vm.get_interned(*i).to_string()),
1360                            _ => {}
1361                        }
1362                    }
1363                    query = query.with_fields(fields);
1364                }
1365
1366                // Check for namespace filter
1367                if let Some(ns_val) = map.get(vm, ns_key) {
1368                    match ns_val {
1369                        Value::String(s) => {
1370                            query = query.with_namespace(vm.get_string(s).to_string())
1371                        }
1372                        Value::StringConst(i) => {
1373                            query = query.with_namespace(vm.get_interned(i).to_string())
1374                        }
1375                        _ => {}
1376                    }
1377                }
1378
1379                // Check for section filter
1380                if let Some(section_val) = map.get(vm, section_key) {
1381                    match section_val {
1382                        Value::String(s) => {
1383                            query = query.with_section(vm.get_string(s).to_string())
1384                        }
1385                        Value::StringConst(i) => {
1386                            query = query.with_section(vm.get_interned(i).to_string())
1387                        }
1388                        _ => {}
1389                    }
1390                }
1391
1392                // Check for styled option
1393                if let Some(Value::False) = map.get(vm, styled_key) {
1394                    use_styled = false;
1395                }
1396
1397                // Check for style customization
1398                if let Some(Value::Map(style_handle)) = map.get(vm, style_key) {
1399                    use_styled = true; // Enable styling if style options are provided
1400                    style_map_handle_opt = Some(style_handle);
1401                }
1402            }
1403            _ => {
1404                return Err(VMError::new_vm(
1405                    "Second argument must be a map of options".to_string(),
1406                ));
1407            }
1408        }
1409    }
1410
1411    // Process style options after we're done with the initial map
1412    if let Some(style_handle) = style_map_handle_opt {
1413        style_options = StyleOptions::from_handle(vm, style_handle);
1414    }
1415
1416    // Check if markdown output is requested
1417    let mut use_markdown = false;
1418    if registers.len() > 1 {
1419        if let Value::Map(handle) = &registers[1] {
1420            let markdown_key = Value::Keyword(vm.intern_static("markdown"));
1421            let map = vm.get_map(*handle);
1422            if let Some(Value::True) = map.get(vm, markdown_key) {
1423                use_markdown = true;
1424            }
1425        }
1426    }
1427
1428    let results = search_docs(vm, &query, false);
1429
1430    let result = if use_markdown {
1431        // Generate markdown output
1432        let mut markdown_output = String::new();
1433        let len = results.len();
1434        for (i, doc) in results.iter().enumerate() {
1435            // Don't print to stdout when generating markdown - the numbered list is included in the markdown output instead
1436            if i > 0 {
1437                markdown_output.push_str("\n---\n\n");
1438            }
1439            // Add a header showing which result this is
1440            markdown_output.push_str(&format!(
1441                "**Result {}: `{}::{}`**\n\n",
1442                i + 1,
1443                doc.namespace,
1444                doc.symbol
1445            ));
1446            markdown_output.push_str(&doc.as_md());
1447        }
1448        markdown_output.push_str(&format!(
1449            "\nFound {len} result{}.",
1450            if len == 1 { "" } else { "s" }
1451        ));
1452        vm.alloc_string(markdown_output)
1453    } else if use_styled {
1454        // Generate styled string output
1455        let styled_output = generate_styled_output(vm, &results, &query, &style_options);
1456        vm.alloc_string(styled_output)
1457    } else {
1458        // Generate vector of doc maps
1459        let mut result_values = Vec::new();
1460        for doc in results {
1461            match BridgeError::with_fn(Value::sl_from(doc, vm), "doc-search") {
1462                Ok(val) => result_values.push(val),
1463                Err(e) => {
1464                    vm.unpause_gc();
1465                    return Err(e);
1466                }
1467            }
1468        }
1469        vm.alloc_vector(result_values)
1470    };
1471
1472    Ok(result)
1473}
