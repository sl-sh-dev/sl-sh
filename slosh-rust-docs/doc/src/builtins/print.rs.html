9fn is_sym(vm: &SloshVm, name: &str, intern: Interned) -> bool {
10    if let Some(i) = vm.get_if_interned(name) {
11        if intern == i {
12            return true;
13        }
14    }
15    false
16}

18fn quotey(vm: &SloshVm, car: Value, buf: &mut String) -> bool {
19    if let Value::Symbol(i) = car {
20        if is_sym(vm, "quote", i) {
21            buf.push('\'');
22            true
23        } else if is_sym(vm, "back-quote", i) {
24            buf.push('`');
25            true
26        } else if is_sym(vm, "unquote", i) {
27            buf.push(',');
28            true
29        } else if is_sym(vm, "unquote-splice", i) {
30            buf.push_str(",@");
31            true
32        } else if is_sym(vm, "unquote-splice!", i) {
33            buf.push_str(",.");
34            true
35        } else {
36            false
37        }
38    } else {
39        false
40    }
41}

43fn list_out(vm: &SloshVm, res: &mut String, lst: Value) {
44    let mut first = true;
45    let mut cdr = lst;
46    loop {
47        if let Value::Nil = cdr {
48            break;
49        }
50        if !first {
51            res.push(' ');
52        } else {
53            first = false;
54        }
55        match cdr {
56            Value::Pair(_) | Value::List(_, _) => {
57                let (car, ncdr) = cdr.get_pair(vm).expect("pair/list not a pair/list");
58                res.push_str(&display_value(vm, car));
59                cdr = ncdr;
60            }
61            _ => {
62                res.push_str(". ");
63                res.push_str(&display_value(vm, cdr));
64                break;
65            }
66        }
67    }
68}

70pub fn display_value(vm: &SloshVm, val: Value) -> String {
71    match &val {
72        Value::Pair(_) | Value::List(_, _) => {
73            let (car, cdr) = val.get_pair(vm).expect("pair/list not a pair/list");
74            let mut res = String::new();
75            if quotey(vm, car, &mut res) {
76                if let Some((cadr, Value::Nil)) = cdr.get_pair(vm) {
77                    res.push_str(&display_value(vm, cadr));
78                } else {
79                    res.push_str(&display_value(vm, cdr));
80                }
81            } else {
82                res.push('(');
83                list_out(vm, &mut res, val);
84                res.push(')');
85            }
86            res
87        }
88        _ => val.display_value(vm),
89    }
90}

92pub fn pretty_value(vm: &SloshVm, val: Value) -> String {
93    match &val {
94        Value::StringConst(i) => vm.get_interned(*i).to_string(),
95        Value::CodePoint(ch) => format!("{ch}"),
96        Value::CharCluster(l, c) => {
97            format!("{}", String::from_utf8_lossy(&c[0..*l as usize]))
98        }
99        Value::CharClusterLong(h) => vm.get_string(*h).to_string(),
100        Value::String(h) => vm.get_string(*h).to_string(),
101        _ => display_value(vm, val),
102    }
103}

129pub fn fpr(vm: &mut SloshVm, registers: &[Value]) -> VMResult<Value> {
130    let mut args = registers.iter();
131    if let Some(Value::Io(h)) = args.next() {
132        let mut file = vm.get_io(*h).get_io();
133        for v in args {
134            write!(file, "{}", pretty_value(vm, *v))?;
135        }
136        Ok(Value::Nil)
137    } else {
138        Err(VMError::new(
139            "io",
140            "fpr: require a writable IO object as first parameter",
141        ))
142    }
143}

185pub fn dasm(vm: &mut SloshVm, registers: &[Value]) -> VMResult<Value> {
186    if registers.len() != 1 {
187        return Err(VMError::new_compile(
188            "dasm: wrong number of args, expected one",
189        ));
190    }
191    let exp = registers[0].unref(vm);
192    match exp {
193        Value::Lambda(handle) => {
194            let l = vm.get_lambda(handle);
195            l.disassemble_chunk(vm, 0)?;
196            Ok(Value::Nil)
197        }
198        Value::Closure(handle) => {
199            let (l, _) = vm.get_closure(handle);
200            l.disassemble_chunk(vm, 0)?;
201            Ok(Value::Nil)
202        }
203        Value::List(_handle, _start_idx) => {
204            let mut state = CompileState::new_state("", 1, None);
205            pass1(vm, &mut state, exp)?;
206            compile(vm, &mut state, exp, 0)?;
207            state.chunk.disassemble_chunk(vm, 0)?;
208            Ok(Value::Nil)
209        }
210        Value::Pair(_handle) => {
211            let mut state = CompileState::new_state("", 1, None);
212            pass1(vm, &mut state, exp)?;
213            compile(vm, &mut state, exp, 0)?;
214            state.chunk.disassemble_chunk(vm, 0)?;
215            Ok(Value::Nil)
216        }
217        _ => Err(VMError::new_vm("DASM: Not a callable.")),
218    }
219}
